215  virtual StateType performIntegrationStep( const TimeStepType stepSize )
216  {
217  if( !( stepSize == stepSize) )
218  {
219  throw std::runtime_error( "Error in BS integrator, step size is NaN" );
220  }
221 
222  StateType stateAtFirstPoint_;
223  StateType stateAtCenterPoint_;
224  StateType stateAtLastPoint_;
225 
226  bool stepSuccessful = 0;
227 
228  // Compute sub steps to take.
229  for ( unsigned int p = 0; p < subSteps_.size( ); p++ )
230  {
231  subSteps_.at( p ) = stepSize / static_cast< double >(
232  sequence_.at( p ) );
233  }
234 
235  double errorScaleTerm = TUDAT_NAN;
236  for ( unsigned int i = 0; i <= maximumStepIndex_; i++ )
237  {
238  // Compute Euler step and set as state at center point for use with mid-point method.
239  stateAtCenterPoint_ = currentState_ + subSteps_.at( i )
240  * this->stateDerivativeFunction_( currentIndependentVariable_, currentState_ );
241 
242  // Apply modified mid-point rule.
243  stateAtFirstPoint_ = currentState_;
244  IndependentVariableType independentVariableAtFirstPoint_ = currentIndependentVariable_;
245  for ( unsigned int j = 0; j < sequence_.at( i ) - 1; j++ )
246  {
247  stateAtLastPoint_ = executeMidPointMethod(
248  stateAtFirstPoint_, stateAtCenterPoint_,
249  independentVariableAtFirstPoint_, subSteps_.at( i ) );
250 
251  if ( j < sequence_.at( i ) - 2 )
252  {
253  stateAtFirstPoint_ = stateAtCenterPoint_;
254  stateAtCenterPoint_ = stateAtLastPoint_;
255  independentVariableAtFirstPoint_ += subSteps_.at( i );
256  }
257  }
258 
259  // Apply end-point correction.
260  integratedStates_[ i ][ 0 ]
261  = 0.5 * ( stateAtLastPoint_ + stateAtCenterPoint_+ subSteps_.at( i ) * this->stateDerivativeFunction_(
262  currentIndependentVariable_ + stepSize, stateAtLastPoint_ ) );
263 
264  for ( unsigned int k = 1; k < i + 1; k++ )
265  {
266  integratedStates_[ i ][ k ] =
267  integratedStates_[ i ][ k - 1 ] + 1.0 /
268  ( pow( subSteps_.at( i - k ), 2.0 ) / std::pow( subSteps_.at( i ), 2.0 ) - 1.0 )
269  * ( integratedStates_[ i ][ k - 1 ] - integratedStates_[ i - 1 ][ k - 1 ] );
270  }
271 
272  if( i == maximumStepIndex_ )
273  {
274  const StateType errorTolerance_ =
275  ( integratedStates_.at( i ).at( i ).array( ).abs( ) * relativeErrorTolerance_.array( ) ).matrix( )
276  + absoluteErrorTolerance_;
277 
278  double maximumAllowableErrorValue = errorTolerance_.array( ).maxCoeff( );
279  double maximumErrorValue = ( integratedStates_.at( i ).at( i ) - integratedStates_.at( i ).at( i - 1 ) ).array( ).abs( ).maxCoeff( );
280 
281  errorScaleTerm = safetyFactorForNextStepSize_ * std::pow( maximumAllowableErrorValue / maximumErrorValue,
282  ( 1.0 / static_cast< double >( 2 * i - 1 ) ) );
283 
284  if( maximumErrorValue < maximumAllowableErrorValue )
285  {
286  // Accept the current step.
287  lastIndependentVariable_ = currentIndependentVariable_;
288  lastState_ = currentState_;
289  currentIndependentVariable_ += stepSize;
290  currentState_ = integratedStates_[ i ][ i ];
291  stepSize_ = stepSize;
292  stepSuccessful = true;
293  }
294  else
295  {
296  stepSuccessful = false;
297  }
298  }
299  }
300 
301  if( !stepSuccessful )
302  {
303  if( safetyFactorForNextStepSize_ * errorScaleTerm < minimumFactorDecreaseForNextStepSize_ )
304  {
305  this->stepSize_ = stepSize * minimumFactorDecreaseForNextStepSize_ * errorScaleTerm;
306  }
307  else
308  {
309  this->stepSize_ = stepSize * safetyFactorForNextStepSize_ * errorScaleTerm;
310  }
311 
312  if( std::fabs( stepSize_ ) < std::fabs( minimumStepSize_ ) )
313  {
314  isMinimumStepSizeViolated_ = true;
315  throw std::runtime_error( "Error in BS integrator, minimum step size exceeded" );
316  }
317  performIntegrationStep( stepSize_ );
318  }
319  else
320  {
321  if( errorScaleTerm > maximumFactorIncreaseForNextStepSize_ )
322  {
323  this->stepSize_ = stepSize * maximumFactorIncreaseForNextStepSize_;
324  }
325  else
326  {
327  this->stepSize_ = stepSize * errorScaleTerm;
328  }
329 
330  if( std::fabs( this->stepSize_ ) >= std::fabs( maximumStepSize_ ) )
331  {
332  this->stepSize_ = stepSize / ( std::fabs( stepSize ) ) * maximumStepSize_ ;
333  }
334 
335  }
336 
337 
338  return currentState_;
339  }
