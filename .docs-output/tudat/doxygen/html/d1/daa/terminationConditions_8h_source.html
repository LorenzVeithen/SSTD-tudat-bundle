49 inline bool checkMaximumIterationsExceeded( const unsigned int numberOfIterations,
50  const unsigned int maximumNumberOfIterations,
51  const MaximumIterationHandling maximumIterationHandling )
52 {
53  // Check if maximum number of iterations have been exceeded.
54  const bool isMaximumIterationsExceeded = numberOfIterations > maximumNumberOfIterations;
55 
56  // If exceeded, and flag is set to throw run-time exception, proceed.
57  if ( isMaximumIterationsExceeded )
58  {
59  static std::string errorMessage
60  = "Root-finder did not converge within maximum number of iterations!";
61  switch( maximumIterationHandling )
62  {
63  case accept_result:
64  break;
65  case accept_result_with_warning:
66  std::cerr<<errorMessage<<std::endl;
67  break;
68  case throw_exception:
69  throw std::runtime_error( errorMessage );
70  break;
71  }
72 
73  }
74 
75  // Else, simply return whether maximum number of iterations have been exceeded.
76  return isMaximumIterationsExceeded;
77 }

377 template< typename DataType = double >
378 std::shared_ptr< TerminationCondition< DataType > > createTerminationCondition(
379  const DataType relativeIndependentVariableTolerance = TUDAT_NAN,
380  const DataType absoluteIndependentVariableTolerance = TUDAT_NAN,
381  const DataType rootFunctionTolerance = TUDAT_NAN,
382  const unsigned int maximumNumberOfIterations = 1000,
383  const MaximumIterationHandling maximumIterationHandling = throw_exception )
384 {
385  std::vector< std::shared_ptr< TerminationCondition< DataType > > > terminationConditionList;
386 
387 
388  terminationConditionList.push_back(
389  std::make_shared< MaximumIterationsTerminationCondition< DataType > >(
390  maximumNumberOfIterations, maximumIterationHandling ) );
391 
392  if( ( relativeIndependentVariableTolerance == relativeIndependentVariableTolerance ) )
393  {
394  terminationConditionList.push_back(
395  std::make_shared< RootRelativeToleranceTerminationCondition< DataType > >(
396  relativeIndependentVariableTolerance ) );
397  }
398 
399  if( ( absoluteIndependentVariableTolerance == absoluteIndependentVariableTolerance ) )
400  {
401  terminationConditionList.push_back(
402  std::make_shared< RootAbsoluteToleranceTerminationCondition< DataType > >(
403  absoluteIndependentVariableTolerance ) );
404  }
405 
406  if( ( rootFunctionTolerance == rootFunctionTolerance ) )
407  {
408  terminationConditionList.push_back(
409  std::make_shared< RootFunctionTerminationCondition< DataType > >(
410  absoluteIndependentVariableTolerance ) );
411  }
412 
413  if( terminationConditionList.size( ) == 1 )
414  {
415  return terminationConditionList.at( 0 );
416  }
417  else
418  {
419  return std::make_shared< CombinedTerminationCondition< DataType > >(
420  terminationConditionList );
421  }
422 }
