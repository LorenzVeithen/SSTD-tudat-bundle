122  DataType execute( const FunctionPointer rootFunction, const DataType initialGuess = 0.0 )
123  {
124  // The value of the initialGuess is not used.
125  TUDAT_UNUSED_PARAMETER( initialGuess );
126 
127  // Set the root function.
128  this->rootFunction = rootFunction;
129 
130  // Initialize previous values.
131  DataType previousRootValue = TUDAT_NAN;
132  DataType previousRootFunctionValue = TUDAT_NAN;
133 
134  // Duplicate the interval and use this duplicate to shrink the interval.
135  // Compute the midpoint of the interval, and take this as the current root.
136  DataType currentLowerBound = lowerBound_;
137  DataType currentUpperBound = upperBound_;
138  DataType rootValue = ( currentLowerBound + currentUpperBound ) / 2.0;
139 
140  // Find the corresponding function values at the important interval points (lower bound,
141  // upper bound and midpoint).
142  DataType currentLowerBoundFunctionValue = this->rootFunction->evaluate( currentLowerBound );
143  DataType currentUpperBoundFunctionValue = this->rootFunction->evaluate( currentUpperBound );
144  if( currentLowerBoundFunctionValue == mathematical_constants::getFloatingInteger< DataType >( 0 ) )
145  {
146  rootValue = currentLowerBound;
147  }
148  else if( currentUpperBoundFunctionValue == mathematical_constants::getFloatingInteger< DataType >( 0 ) )
149  {
150  rootValue = currentUpperBound;
151  }
152  else
153  {
154  DataType rootFunctionValue = this->rootFunction->evaluate( rootValue );
155 
156  // Validate that upperbound and lowerbound function values have different signs
157  // (requirement).
158  if( currentLowerBoundFunctionValue * currentUpperBoundFunctionValue > 0.0 )
159  {
160  throw std::runtime_error( "The Bisection algorithm requires that the values at the upper, "
161  "and lower bounds have a different sign." );
162  }
163 
164  // Loop counter.
165  unsigned int counter = 1;
166 
167  // Loop until we have a solution with sufficient accuracy.
168  do
169  {
170  // Sanity check.
171  if( currentLowerBoundFunctionValue * currentUpperBoundFunctionValue > 0.0 )
172  {
173  throw std::runtime_error( "The Bisection algorithm requires that the values at the upper, "
174  "and lower bounds have a different sign, error during iteration." );
175  }
176 
177  // Save old values.
178  previousRootValue = rootValue;
179  previousRootFunctionValue = rootFunctionValue;
180 
181  // Check which subinterval to keep, by maintaining endpoints with opposite function
182  // value signs.
183  if( rootFunctionValue * currentLowerBoundFunctionValue < 0.0 )
184  {
185  // Different sign, hence the upper bound is replaced.
186  currentUpperBound = rootValue;
187  currentUpperBoundFunctionValue = rootFunctionValue;
188  }
189  else
190  {
191  // Same sign, hence the lower bound is replaced.
192  currentLowerBound = rootValue;
193  currentLowerBoundFunctionValue = rootFunctionValue;
194  }
195 
196  // Compute the new midpoint of the interval and its function value.
197  rootValue = ( currentLowerBound + currentUpperBound ) / 2.0;
198  rootFunctionValue = this->rootFunction->evaluate( rootValue );
199 
200  counter++;
201  }
202  while( rootFunctionValue != mathematical_constants::getFloatingInteger< DataType >( 0 ) &&
203  !this->terminationFunction_( rootValue, previousRootValue, rootFunctionValue,
204  previousRootFunctionValue, counter ) );
205  }
206  return rootValue;
207  }
