139 template< typename ObservationScalarType = double, typename TimeType = double,
140  int ObservationSize = 1 >
141 std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,std::pair< std::vector< TimeType >, LinkEndType > >
142 simulateSingleObservationSet(
143  const std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > observationsToSimulate,
144  const std::shared_ptr< ObservationModel< ObservationSize, ObservationScalarType, TimeType > > observationModel,
145  const std::vector< std::shared_ptr< ObservationViabilityCalculator > > currentObservationViabilityCalculators =
146  std::vector< std::shared_ptr< ObservationViabilityCalculator > >( ) )
147 {
148  // Delcare return type.
149  std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >, std::pair< std::vector< TimeType >, LinkEndType > >
150  simulatedObservations;
151 
152  // Simulate observations from tabulated times.
153  if( std::dynamic_pointer_cast< TabulatedObservationSimulationTimeSettings< TimeType > >( observationsToSimulate ) != nullptr )
154  {
155  std::shared_ptr< TabulatedObservationSimulationTimeSettings< TimeType > > tabulatedObservationSettings =
156  std::dynamic_pointer_cast< TabulatedObservationSimulationTimeSettings< TimeType > >( observationsToSimulate );
157 
158  // Simulate observations at requested pre-defined time.
159  simulatedObservations = simulateObservationsWithCheckAndLinkEndIdOutput<
160  ObservationSize, ObservationScalarType, TimeType >(
161  tabulatedObservationSettings->simulationTimes_, observationModel, observationsToSimulate->linkEndType_,
162  currentObservationViabilityCalculators );
163 
164  }
165  // Simulate observations per arc from settings
166  else if( std::dynamic_pointer_cast< ArcLimitedObservationSimulationTimeSettings< TimeType > >( observationsToSimulate ) != NULL )
167  {
168  std::shared_ptr< ArcLimitedObservationSimulationTimeSettings< TimeType > > arcLimitedObservationSettings =
169  std::dynamic_pointer_cast< ArcLimitedObservationSimulationTimeSettings< TimeType > >( observationsToSimulate );
170 
171  // Define constituents of return pair.
172  std::vector< TimeType > times;
173  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > observations;
174  int observableSize = observationModel->getObservationSize( );
175 
176  // Define vector to be used for storing single arc observations.
177  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > singleArcObservations =
178  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >::Zero(
179  observableSize * arcLimitedObservationSettings->observationLimitPerArc_ );
180 
181  // Set start and end times.
182  TimeType currentTime = arcLimitedObservationSettings->startTime_;
183  TimeType currentArcStartTime = arcLimitedObservationSettings->startTime_;
184 
185  int totalNumberOfObservations = 0;
186  int numberOfObservationsInCurrentArc = 0;
187 
188  // Simulate observations arcs until provided end time.
189  while( currentTime < arcLimitedObservationSettings->endTime_ - 0.1 )
190  {
191  // Reset variables for start of new arc.
192  numberOfObservationsInCurrentArc = 0;
193  singleArcObservations.setZero( );
194 
195  // Simulate observations for sinlge arc
196  while( ( currentTime < currentArcStartTime + arcLimitedObservationSettings->arcDuration_ ) &&
197  ( numberOfObservationsInCurrentArc < arcLimitedObservationSettings->observationLimitPerArc_ ) &&
198  ( currentTime < arcLimitedObservationSettings->endTime_ ) )
199  {
200  // Simulate single observation
201  std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >, bool > currentObservation =
202  simulateObservationWithCheck<
203  ObservationSize, ObservationScalarType, TimeType >(
204  currentTime, observationModel, observationsToSimulate->linkEndType_,
205  currentObservationViabilityCalculators );
206 
207  // If observation is possible, set it in current arc observations.
208  if( currentObservation.second )
209  {
210  times.push_back( currentTime );
211  singleArcObservations.segment( numberOfObservationsInCurrentArc * observableSize, observableSize ) = currentObservation.first;
212  numberOfObservationsInCurrentArc += observableSize;
213  }
214  currentTime += arcLimitedObservationSettings->observationInterval_;
215  }
216 
217  // Add single arc observations to total observations.
218  observations.conservativeResize( totalNumberOfObservations + numberOfObservationsInCurrentArc );
219  observations.segment( totalNumberOfObservations, numberOfObservationsInCurrentArc ) = singleArcObservations.segment( 0, numberOfObservationsInCurrentArc );
220  totalNumberOfObservations += numberOfObservationsInCurrentArc;
221 
222  // Update times to next arc
223  currentArcStartTime += arcLimitedObservationSettings->arcDuration_;
224  currentTime = currentArcStartTime;
225 
226  }
227 
228  simulatedObservations = std::make_pair( observations, std::make_pair( times, arcLimitedObservationSettings->linkEndType_ ) );
229  }
230 
231  return simulatedObservations;
232 }

348 template< typename ObservationScalarType = double, typename TimeType = double >
349 std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
350 std::pair< std::vector< TimeType >, LinkEndType > > > >
351 simulateObservations(
352  const std::map< ObservableType, std::map< LinkEnds,
353  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >& observationsToSimulate,
354  const std::map< ObservableType,
355  std::shared_ptr< ObservationSimulatorBase< ObservationScalarType, TimeType > > >& observationSimulators,
356  const PerObservableObservationViabilityCalculatorList viabilityCalculatorList =
357  PerObservableObservationViabilityCalculatorList( ) )
358 {
359  // Declare return map.
360  std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
361  std::pair< std::vector< TimeType >, LinkEndType > > > > observations;
362 
363  // Iterate over all observables.
364  for( typename std::map< ObservableType, std::map< LinkEnds,
365  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >::const_iterator observationIterator =
366  observationsToSimulate.begin( ); observationIterator != observationsToSimulate.end( ); observationIterator++ )
367  {
368 
369  PerLinkEndsObservationViabilityCalculatorList perLinkViabilityCalculators;
370  if( viabilityCalculatorList.count( observationIterator->first ) > 0 )
371  {
372  perLinkViabilityCalculators = viabilityCalculatorList.at( observationIterator->first );
373  }
374  // Iterate over all link ends for current observable.
375  for( typename std::map< LinkEnds,
376  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > >::const_iterator linkEndIterator =
377  observationIterator->second.begin( ); linkEndIterator != observationIterator->second.end( ); linkEndIterator++ )
378  {
379 
380  std::vector< std::shared_ptr< ObservationViabilityCalculator > > currentObservationViabilityCalculators;
381  if( perLinkViabilityCalculators.count( linkEndIterator->first ) > 0 )
382  {
383  currentObservationViabilityCalculators = perLinkViabilityCalculators.at( linkEndIterator->first );
384  }
385 
386  int observationSize = observationSimulators.at( observationIterator->first )->getObservationSize(
387  linkEndIterator->first );
388 
389  switch( observationSize )
390  {
391  case 1:
392  {
393  std::shared_ptr< ObservationSimulator< 1, ObservationScalarType, TimeType > > derivedObservationSimulator =
394  std::dynamic_pointer_cast< ObservationSimulator< 1, ObservationScalarType, TimeType > >(
395  observationSimulators.at( observationIterator->first ) );
396 
397  if( derivedObservationSimulator == nullptr )
398  {
399  throw std::runtime_error( "Error when simulating observation: dynamic case to size 1 is nullptr" );
400  }
401 
402  // Simulate observations for current observable and link ends set.
403  observations[ observationIterator->first ][ linkEndIterator->first ] =
404  simulateSingleObservationSet< ObservationScalarType, TimeType, 1 >(
405  linkEndIterator->second, derivedObservationSimulator,
406  linkEndIterator->first, currentObservationViabilityCalculators );
407  break;
408  }
409  case 2:
410  {
411  std::shared_ptr< ObservationSimulator< 2, ObservationScalarType, TimeType > > derivedObservationSimulator =
412  std::dynamic_pointer_cast< ObservationSimulator< 2, ObservationScalarType, TimeType > >(
413  observationSimulators.at( observationIterator->first ) );
414 
415  if( derivedObservationSimulator == nullptr )
416  {
417  throw std::runtime_error( "Error when simulating observation: dynamic case to size 2 is nullptr" );
418  }
419 
420  // Simulate observations for current observable and link ends set.
421  observations[ observationIterator->first ][ linkEndIterator->first ] =
422  simulateSingleObservationSet< ObservationScalarType, TimeType, 2 >(
423  linkEndIterator->second, derivedObservationSimulator,
424  linkEndIterator->first, currentObservationViabilityCalculators );
425  break;
426  }
427  case 3:
428  {
429  std::shared_ptr< ObservationSimulator< 3, ObservationScalarType, TimeType > > derivedObservationSimulator =
430  std::dynamic_pointer_cast< ObservationSimulator< 3, ObservationScalarType, TimeType > >(
431  observationSimulators.at( observationIterator->first ) );
432 
433  if( derivedObservationSimulator == nullptr )
434  {
435  throw std::runtime_error( "Error when simulating observation: dynamic case to size 3 is nullptr" );
436  }
437 
438  // Simulate observations for current observable and link ends set.
439  observations[ observationIterator->first ][ linkEndIterator->first ] =
440  simulateSingleObservationSet< ObservationScalarType, TimeType, 3 >(
441  linkEndIterator->second, derivedObservationSimulator,
442  linkEndIterator->first, currentObservationViabilityCalculators );
443  break;
444  }
445  default:
446  throw std::runtime_error( "Error, simulation of observations not yet implemented for size " +
447  std::to_string( observationSize ) );
448 
449  }
450  }
451  }
452  return observations;
453 }

468 template< typename ObservationScalarType = double, typename TimeType = double >
469 std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
470 std::pair< std::vector< TimeType >, LinkEndType > > > >
471 simulateObservationsWithNoise(
472  const std::map< ObservableType, std::map< LinkEnds,
473  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >& observationsToSimulate,
474  const std::map< ObservableType,
475  std::shared_ptr< ObservationSimulatorBase< ObservationScalarType, TimeType > > >& observationSimulators,
476  const std::map< ObservableType, std::map< LinkEnds, std::function< Eigen::VectorXd( const double ) > > >& noiseFunctions,
477  const PerObservableObservationViabilityCalculatorList viabilityCalculatorList =
478  PerObservableObservationViabilityCalculatorList( ) )
479 {
480  typedef std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
481  std::pair< std::vector< TimeType >, LinkEndType > > > SingelTypeObservationsMap;
482  typedef std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
483  std::pair< std::vector< TimeType >, LinkEndType > > > > ObservationsMap;
484 
485  // Simulate noise-free observations
486  ObservationsMap noiseFreeObservationsList = simulateObservations(
487  observationsToSimulate, observationSimulators, viabilityCalculatorList );
488 
489  // Declare return map with noisy observations.
490  ObservationsMap noisyObservationsList;
491 
492  // Iterate over all observable types
493  for( typename ObservationsMap::iterator observationIterator = noiseFreeObservationsList.begin( );
494  observationIterator != noiseFreeObservationsList.end( ); observationIterator++ )
495  {
496  int currentObservableSize = getObservableSize( observationIterator->first );
497 
498  // Iterate over all link ends of current observable
499  SingelTypeObservationsMap singleObservableObservationsWithNoise;
500  for( typename SingelTypeObservationsMap::iterator linkIterator = observationIterator->second.begin( );
501  linkIterator != observationIterator->second.end( ); linkIterator++ )
502  {
503  // Retrieve noise-free observations/times
504  std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
505  std::pair< std::vector< TimeType >, LinkEndType > > nominalObservationsAndTimes =
506  linkIterator->second;
507  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > noisyObservations = nominalObservationsAndTimes.first;
508  std::vector< TimeType > nominalTimes = nominalObservationsAndTimes.second.first;
509 
510  // Check data consistency
511  if( static_cast< int >( nominalTimes.size( ) ) != noisyObservations.rows( ) / currentObservableSize )
512  {
513  throw std::runtime_error( "Error when adding noise to observations, input data is inconsistent" );
514  }
515 
516  // Retrieve noise function
517  std::function< Eigen::VectorXd( const double ) > currentNoiseFunction =
518  noiseFunctions.at( observationIterator->first ).at( linkIterator->first );
519 
520  // Simulate noise for all observations and add to calculated values
521  for( unsigned int i = 0; i < nominalTimes.size( ); i++ )
522  {
523  // Check noise function consistency
524  if( i == 0 )
525  {
526  if( currentNoiseFunction( nominalTimes.at( i ) ).rows( ) != currentObservableSize )
527  {
528  throw std::runtime_error( "Error when adding noise to observations, noise size is inconsistent" );
529  }
530  }
531 
532  // Add noise to observation
533  noisyObservations.segment( i * currentObservableSize, currentObservableSize ) +=
534  currentNoiseFunction( nominalTimes.at( i ) );
535  }
536 
537  singleObservableObservationsWithNoise[ linkIterator->first ] =
538  std::make_pair( noisyObservations, nominalObservationsAndTimes.second );
539  }
540  noisyObservationsList[ observationIterator->first ] = singleObservableObservationsWithNoise;
541  }
542  return noisyObservationsList;
543 }
