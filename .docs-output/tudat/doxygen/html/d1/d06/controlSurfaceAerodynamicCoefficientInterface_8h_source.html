52  ControlSurfaceIncrementAerodynamicInterface(
53  const std::vector< AerodynamicCoefficientsIndependentVariables > independentVariableNames ):
54  independentVariableNames_( independentVariableNames )
55  {
56  // Count number of control surface deflections.
57  int numberOfControlSurfaceDeflections = 0;
58  for( unsigned int i = 0; i < independentVariableNames_.size( ); i++ )
59  {
60  if( independentVariableNames_.at( i ) == control_surface_deflection_dependent )
61  {
62  numberOfControlSurfaceDeflections++;
63  }
64  }
65 
66  // Provide warning if number of control surface deflections is not equal to 1.
67  if( numberOfControlSurfaceDeflections != 1 )
68  {
69  std::string errorMessage = "Warnining when making control surface deflection interface, " + std::to_string(
70  numberOfControlSurfaceDeflections ) + " deflection independent variables found, must be 1 per object ";
71  throw std::runtime_error( errorMessage );
72  }
73 
74  numberOfIndependentVariables_ = independentVariableNames_.size( );
75  }
