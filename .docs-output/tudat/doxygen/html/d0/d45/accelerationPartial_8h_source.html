68  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
69  getDerivativeFunctionWrtStateOfIntegratedBody(
70  const std::pair< std::string, std::string >& stateReferencePoint,
71  const propagators::IntegratedStateType integratedStateType )
72  {
73  // Initialize to empty function; 0 parameter size.
74  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
75  partialFunction = std::make_pair( std::function< void( Eigen::Block< Eigen::MatrixXd > ) >( ), 0 );
76 
77  // Check if state dependency exists
78  switch( integratedStateType )
79  {
80  case propagators::translational_state:
81  {
82  // Check if reference id is consistent.
83  if( stateReferencePoint.second != "" )
84  {
85  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for dynamics" );
86  }
87  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
88  else if( stateReferencePoint.first == acceleratedBody_ )
89  {
90  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAcceleratedBody, this, std::placeholders::_1 ), 3 );
91  }
92  else if( stateReferencePoint.first == acceleratingBody_ )
93  {
94  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAcceleratingBody, this, std::placeholders::_1 ), 3 );
95  }
96  else if( isAccelerationPartialWrtAdditionalBodyNonnullptr( stateReferencePoint.first ) )
97  {
98  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAdditionalBody,
99  this, std::placeholders::_1, stateReferencePoint.first ), 3 );
100  }
101  break;
102  }
103  case propagators::rotational_state:
104  {
105  // Check if reference id is consistent.
106  if( stateReferencePoint.second != "" )
107  {
108  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for body mass" );
109  }
110  else if( isStateDerivativeDependentOnIntegratedAdditionalStateTypes( stateReferencePoint, integratedStateType ) )
111  {
112  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtNonTranslationalStateOfAdditionalBody,
113  this, std::placeholders::_1, stateReferencePoint, integratedStateType, true ), 1 );
114  }
115  break;
116  }
117  case propagators::body_mass_state:
118  {
119  // Check if reference id is consistent.
120  if( stateReferencePoint.second != "" )
121  {
122  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for body mass" );
123  }
124  else if( isStateDerivativeDependentOnIntegratedAdditionalStateTypes( stateReferencePoint, integratedStateType ) )
125  {
126  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtNonTranslationalStateOfAdditionalBody,
127  this, std::placeholders::_1, stateReferencePoint, integratedStateType, true ), 1 );
128  }
129  break;
130  }
131  case propagators::custom_state:
132  {
133  break;
134  }
135  default:
136  std::string errorMessage =
137  "Error when getting state derivative partial acceleration model, dynamics type " +
138  std::to_string( integratedStateType ) + "not recognized" ;
139  throw std::runtime_error( errorMessage );
140  break;
141  }
142 
143 
144  return partialFunction;
145  }

170  bool isStateDerivativeDependentOnIntegratedState(
171  const std::pair< std::string, std::string >& stateReferencePoint,
172  const propagators::IntegratedStateType integratedStateType )
173  {
174  bool isDependent = 0;
175 
176  // Check if state is translational.
177  switch( integratedStateType )
178  {
179  case propagators::translational_state:
180  {
181  // Check if reference id is consistent.
182  if( stateReferencePoint.second != "" )
183  {
184  throw std::runtime_error( "Error when checking state derivative partial dependency of acceleration model, cannot have reference point on body for dynamics" );
185  }
186 
187  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
188  else if( stateReferencePoint.first == acceleratedBody_ || stateReferencePoint.first == acceleratingBody_ ||
189  isAccelerationPartialWrtAdditionalBodyNonnullptr( stateReferencePoint.first ) )
190  {
191  isDependent = 1;
192  }
193  break;
194  }
195  default:
196  isDependent = isStateDerivativeDependentOnIntegratedAdditionalStateTypes( stateReferencePoint, integratedStateType );
197  break;
198  }
199  return isDependent;
200  }
