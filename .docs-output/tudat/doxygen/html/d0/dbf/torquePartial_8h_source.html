84  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
85  getDerivativeFunctionWrtStateOfIntegratedBody(
86  const std::pair< std::string, std::string >& stateReferencePoint,
87  const propagators::IntegratedStateType integratedStateType )
88  {
89  // Initialize to empty function; 0 parameter size.
90  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
91  partialFunction = std::make_pair( std::function< void( Eigen::Block< Eigen::MatrixXd > ) >( ), 0 );
92 
93  // Check if state dependency exists
94  switch( integratedStateType )
95  {
96  case propagators::rotational_state:
97  {
98  // Check if reference id is consistent.
99  if( stateReferencePoint.second != "" )
100  {
101  throw std::runtime_error( "Error when getting state derivative partial torque model, cannot have reference point on body for dynamics" );
102  }
103  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
104  else if( stateReferencePoint.first == bodyUndergoingTorque_ )
105  {
106  partialFunction = std::make_pair( std::bind( &TorquePartial::wrtRotationalStateOfAcceleratedBody, this, std::placeholders::_1 ), 7 );
107  }
108  else if( stateReferencePoint.first == bodyExertingTorque_ )
109  {
110  partialFunction = std::make_pair( std::bind( &TorquePartial::wrtRotationalStateOfAcceleratingBody, this, std::placeholders::_1 ), 7 );
111  }
112  else if( isTorquePartialWrtAdditionalBodyNonNull( stateReferencePoint.first ) )
113  {
114  partialFunction = std::make_pair( std::bind( &TorquePartial::wrtRotationalStateOfAdditionalBody,
115  this, std::placeholders::_1, stateReferencePoint.first ), 3 );
116  }
117  break;
118  }
119  case propagators::translational_state:
120  {
121  // Check if reference id is consistent.
122  if( stateReferencePoint.second != "" )
123  {
124  throw std::runtime_error( "Error when getting torque partial, cannot have reference point on body for body mass" );
125  }
126  else if( isStateDerivativeDependentOnIntegratedAdditionalStateTypes( stateReferencePoint, integratedStateType ) )
127  {
128  partialFunction = std::make_pair( std::bind( &TorquePartial::wrtNonRotationalStateOfAdditionalBody,
129  this, std::placeholders::_1, stateReferencePoint, integratedStateType ), 1 );
130  }
131  }
132  case propagators::custom_state:
133  {
134  break;
135  }
136  default:
137  std::string errorMessage =
138  "Error when getting state derivative partial torque model, dynamics type " +
139  std::to_string( integratedStateType ) + "not recognized" ;
140  throw std::runtime_error( errorMessage );
141  break;
142  }
143 
144 
145  return partialFunction;
146  }
