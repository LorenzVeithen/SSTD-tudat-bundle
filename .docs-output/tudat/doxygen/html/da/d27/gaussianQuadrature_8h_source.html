120  IndependentVariableArray getNodes( const unsigned int order )
121  {
122  if ( nodes_.count( order ) == 0 )
123  {
124  IndependentVariableArray newNodes( order );
125 
126  // Include node 0.0 if order is odd
127  unsigned int i = 0;
128  if ( order % 2 == 1 )
129  {
130  newNodes.row( i++ ) = 0.0;
131  }
132 
133  // Include ± nodes
134  IndependentVariableArray uniqueNodes_ = getUniqueNodes( order );
135  for ( int j = 0; j < uniqueNodes_.size( ); j++ )
136  {
137  newNodes.row( i++ ) = -uniqueNodes_[ j ];
138  newNodes.row( i++ ) = uniqueNodes_[ j ];
139  }
140 
141  nodes_[ order ] = newNodes;
142  }
143 
144  return nodes_.at( order );
145  }
146 
148  IndependentVariableArray getWeights( const unsigned int n )
149  {
150  if ( weights_.count( n ) == 0 )
151  {
152  IndependentVariableArray newWeights( n );
153  IndependentVariableArray orderNWeights = getUniqueWeights( n );
154 
155  // Include non-repeated weight factor if n is odd
156  unsigned int i = 0;
157  int j = 0;
158  if ( n % 2 == 1 )
159  {
160  newWeights.row( i++ ) = orderNWeights[ j++ ];
161  }
162 
163  // Include repeated weight factors
164  for ( ; j < orderNWeights.size( ); j++ )
165  {
166  newWeights.row( i++ ) = orderNWeights[ j ];
167  newWeights.row( i++ ) = orderNWeights[ j ];
168  }
169 
170  weights_[ n ] = newWeights;
171  }
172 
173  return weights_.at( n );
174  }

297  DependentVariableType getQuadrature( )
298  {
299  if ( ! quadratureHasBeenPerformed_ )
300  {

317  }

319  performQuadrature( );

334  void performQuadrature( )
335  {
336  // Determine the values of the auxiliary independent variable (nodes)
337  const IndependentVariableArray nodes = gaussQuadratureNodesAndWeights_->getNodes( numberOfNodes_ );
338 
339  // Determine the values of the weight factors
340  const IndependentVariableArray weights = gaussQuadratureNodesAndWeights_->getWeights( numberOfNodes_ );
341 
342  // Change of variable -> from range [-1, 1] to range [lowerLimit, upperLimit]
343  const IndependentVariableArray independentVariables =
344  0.5 * ( ( upperLimit_ - lowerLimit_ ) * nodes + upperLimit_ + lowerLimit_ );
345 
346  // Determine the value of the dependent variable
347  DependentVariableArray weighedIntegrands( numberOfNodes_ );
348  for ( unsigned int i = 0; i < numberOfNodes_; i++ )
349  {
350  weighedIntegrands( i ) = weights( i ) * integrand_( independentVariables( i ) );
351  }
352 
353  quadratureResult_ = 0.5 * ( upperLimit_ - lowerLimit_ ) * weighedIntegrands.sum( );
354  }
