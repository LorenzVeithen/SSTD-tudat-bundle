316 // Interface class that allows single interpolator to be used for thrust direction and magnitude (which are separated in
317 // thrust implementation)
318 // TODO: not exposed
319 class FullThrustInterpolationInterface
320 {
321 public:
322 
323  // Constructor
330  FullThrustInterpolationInterface(
331  const std::shared_ptr< interpolators::OneDimensionalInterpolator<
332  double, Eigen::Vector3d > > thrustInterpolator,
333  const std::function< Eigen::Matrix3d( ) > rotationFunction =
334  [ ]( ){ return Eigen::Matrix3d::Identity( ); } ):
335  thrustInterpolator_( thrustInterpolator ), rotationFunction_( rotationFunction ),
336  currentThrust_( Eigen::Vector3d::Constant( TUDAT_NAN ) ), currentTime_( TUDAT_NAN ){ }
337 
338  // Function to retrieve the current thrust magnitude
344  double getThrustMagnitude( const double time )
345  {
346  updateThrust( time );
347 
348  return currentThrust_.norm( );
349  }
350 
351  // Function to retrieve the current thrust direction (in the propagation frame).
358  Eigen::Vector3d getThrustDirection( const double time )
359  {
360  updateThrust( time );
361  return currentThrust_.normalized( );
362  }
363 
364  // Function to reset the function to rotate to propation frame
370  void resetRotationFunction( const std::function< Eigen::Matrix3d( ) > rotationFunction )
371  {
372  rotationFunction_ = rotationFunction;
373  }
374 
375  void resetTime( const double currentTime = TUDAT_NAN )
376  {
377  currentTime_ = currentTime;
378  }
379 
380  // Function to retrieve the thrust interpolator.
384  std::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::Vector3d > > getThrustInterpolator( )
385  {
386  return thrustInterpolator_;
387  }
388 
389 private:
390 
391  // Function to update the thrust vector to the current time
396  void updateThrust( const double time )
397  {
398  if( !( time == currentTime_ ) )
399  {
400  currentThrust_ = rotationFunction_( ) * thrustInterpolator_->interpolate( time );
401  currentTime_ = time;
402  }
403  }
404 
405  // Object that returns the total thrust vector, expressed in some reference frame B
406  std::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::Vector3d > > thrustInterpolator_;
407 
408  // Function that returns the rotation matrix from the frame B to the frame in which the propagation is performed.
409  std::function< Eigen::Matrix3d( ) > rotationFunction_;
410 
411  // Total thrust vector (in propagation frame) computed by last call to updateThrust function.
412  Eigen::Vector3d currentThrust_;
413 
414  // Time at which the last call to updateThrust was made (e.g. time associated with current thrust).
415  double currentTime_;
416 
417 };

584 // TODO: not exposed
585 // Retrieve acceleration model (thrust).
586 inline std::shared_ptr< simulation_setup::ThrustAccelerationSettings > getLowThrustLegAccelerationSettings(
587  const std::shared_ptr< low_thrust_trajectories::LowThrustLeg > lowThrustLeg,
588  const simulation_setup::SystemOfBodies& bodies,
589  const std::string& bodyToPropagate,
590  const std::function< double( const double ) > specificImpulseFunction,
591  const double lowThrustLegInitialTime )
592 {
593  using namespace low_thrust_trajectories;
594 
595  std::shared_ptr< simulation_setup::Body > vehicle = bodies.at( bodyToPropagate );
596 
597  // Define thrust magnitude function from the shaped trajectory.
598  std::function< double( const double ) > thrustForceMagnitudeFunction;
599  if( lowThrustLeg->getLegModelIsForceBased( ) )
600  {
601  thrustForceMagnitudeFunction =
602  std::bind( &LowThrustLeg::getForceBasedThrustMagnitude, lowThrustLeg,
603  std::placeholders::_1, lowThrustLegInitialTime );
604  }
605  else
606  {
607  std::function< double( const double, const double ) > thrustAccelerationMagnitudeFunction =
608  std::bind( &LowThrustLeg::getAccelerationBasedThrustMagnitude, lowThrustLeg,
609  std::placeholders::_1, lowThrustLegInitialTime, std::placeholders::_2 );
610  std::function< double( ) > bodyMassFunction =
611  std::bind( &Body::getBodyMass, vehicle );
612  thrustForceMagnitudeFunction = [=](const double currentTime ){
613  return thrustAccelerationMagnitudeFunction( currentTime, bodyMassFunction( ) ); };
614  }
615 
616  // Define thrust magnitude settings from thrust magnitude function.
617  std::shared_ptr< simulation_setup::FromFunctionThrustMagnitudeSettings > thrustMagnitudeSettings =
618  std::make_shared< simulation_setup::FromFunctionThrustMagnitudeSettings >(
619  thrustForceMagnitudeFunction, specificImpulseFunction );
620 
621 
622  // Define thrust direction function from the shaped trajectory.
623  std::function< Eigen::Vector3d( const double ) > thrustDirectionFunction =
624  std::bind( &LowThrustLeg::getThrustDirection, lowThrustLeg,
625  std::placeholders::_1, lowThrustLegInitialTime );
626 
627  // Define thrust direction settings from the direction of thrust acceleration retrieved from the shaping method.
628  std::shared_ptr< simulation_setup::CustomThrustDirectionSettings > thrustDirectionSettings =
629  std::make_shared< simulation_setup::CustomThrustDirectionSettings >( thrustDirectionFunction );
630 
631  // Define thrust acceleration settings.
632  std::shared_ptr< simulation_setup::ThrustAccelerationSettings > thrustAccelerationSettings =
633  std::make_shared< simulation_setup::ThrustAccelerationSettings >(
