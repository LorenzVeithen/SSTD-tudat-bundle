134  DataType execute( const FunctionPointer rootFunction, const DataType initialGuess )
135  {
136  // Set the root function.
137  this->rootFunction = rootFunction;
138 
139  // Start at the two initial values that are used in the algorithm, and compute the function
140  // values.
141  DataType lastRootValue = TUDAT_NAN;
142  DataType currentRootValue = initialGuessOfRootOne_;
143  DataType nextRootValue = initialGuess;
144  DataType lastFunctionValue = TUDAT_NAN;
145  DataType currentFunctionValue = this->rootFunction->evaluate( currentRootValue );
146  DataType nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
147 
148  // Check if the next root value is the most accurate guess. If not, switch the values.
149  if( std::fabs( currentFunctionValue ) < std::fabs( nextFunctionValue ) )
150  {
151  // Switch the root value.
152  DataType temporaryRoot = currentRootValue;
153  currentRootValue = nextRootValue;
154  nextRootValue = temporaryRoot;
155 
156  // Switch the function value
157  DataType temporaryValue = currentFunctionValue;
158  currentFunctionValue = nextFunctionValue;
159  nextFunctionValue = temporaryValue;
160  }
161 
162  // Loop counter.
163  unsigned int counter = 1;
164 
165  // Loop until we have a solution with sufficient accuracy.
166  do
167  {
168  // Save the old values.
169  lastRootValue = currentRootValue;
170  lastFunctionValue = currentFunctionValue;
171  currentRootValue = nextRootValue;
172  currentFunctionValue = nextFunctionValue;
173 
174  // Compute next value of root using the following algorithm (see class documentation):
175  nextRootValue = currentRootValue - currentFunctionValue
176  * ( currentRootValue - lastRootValue )
177  / ( currentFunctionValue - lastFunctionValue );
178  nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
179 
180  // Update the counter.
181  counter++;
182  }
183  while( nextFunctionValue != mathematical_constants::getFloatingInteger< DataType >( 0 ) &&
184  !this->terminationFunction_( nextRootValue, currentRootValue, nextFunctionValue,
185  currentFunctionValue, counter ) );
186 
187  return nextRootValue;
188 
189  }
