68  virtual DependentVariable computeDerivative( const unsigned int order,
69  const IndependentVariable independentVariable )
70  {
71  if ( order == 0 )
72  {
73  // Zero th order derivative is the function itself.
74  return this->evaluate( independentVariable );
75  }
76 
77  // Define step size to the left and right.
78  double stepSize = sqrt_epsilon_double * independentVariable;
79 
80  // Return derivative according to formula:
81  // f'(x) = [f(x+h)-f(x-h)]/2h
82  return ( computeDerivative( order - 1, independentVariable + stepSize )
83  - computeDerivative( order - 1, independentVariable - stepSize ) )
84  / ( 2.0 * stepSize );
85  }
