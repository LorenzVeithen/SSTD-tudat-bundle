277  ObservationScalarType calculateLightTimeWithLinkEndsStates(
278  StateType& receiverStateOutput,
279  StateType& transmitterStateOutput,
280  const TimeType time,
281  const bool isTimeAtReception = 1,
282  const ObservationScalarType tolerance =
283  ( getDefaultLightTimeTolerance< ObservationScalarType >( ) ) )
284  {
285  // Initialize reception and transmission times and states to initial guess (zero light time)
286  TimeType receptionTime = time;
287  TimeType transmissionTime = time;
288  StateType receiverState = stateFunctionOfReceivingBody_( receptionTime );
289  StateType transmitterState =
290  stateFunctionOfTransmittingBody_( transmissionTime );
291 
292  // Set initial light-time correction.
293  setTotalLightTimeCorrection(
294  transmitterState, receiverState, transmissionTime, receptionTime );
295 
296  // Calculate light-time solution assuming infinte speed of signal as initial estimate.
297  ObservationScalarType previousLightTimeCalculation =
298  calculateNewLightTimeEstime( receiverState, transmitterState );
299 
300  // Set variables for iteration
301  ObservationScalarType newLightTimeCalculation = 0.0;
302  bool isToleranceReached = false;
303 
304  // Recalculate light-time solution until tolerance is reached.
305  int counter = 0;
306 
307  // Set variable determining whether to update the light time each iteration.
308  bool updateLightTimeCorrections = false;
309  if( iterateCorrections_ )
310  {
311  updateLightTimeCorrections = true;
312  }
313 
314  // Iterate until tolerance reached.
315  while( !isToleranceReached )
316  {
317  // Update light-time corrections, if necessary.
318  if( updateLightTimeCorrections )
319  {
320  setTotalLightTimeCorrection(
321  transmitterState, receiverState, transmissionTime, receptionTime );
322  }
323 
324  // Update light-time estimate for this iteration.
325  if( isTimeAtReception )
326  {
327  receptionTime = time;
328  transmissionTime = time - previousLightTimeCalculation;
329  transmitterState = ( stateFunctionOfTransmittingBody_( transmissionTime ) );
330  }
331  else
332  {
333  receptionTime = time + previousLightTimeCalculation;
334  transmissionTime = time;
335  receiverState = ( stateFunctionOfReceivingBody_( receptionTime ) );
336  }
337  newLightTimeCalculation = calculateNewLightTimeEstime( receiverState, transmitterState );
338 
339  // Check for convergence.
340  if( std::fabs( newLightTimeCalculation - previousLightTimeCalculation ) < tolerance )
341  {
342  // If convergence reached, but light-time corrections not iterated,
343  // perform 1 more iteration to check for change in correction.
344  if( !updateLightTimeCorrections )
345  {
346  updateLightTimeCorrections = true;
347  }
348  else
349  {
350  isToleranceReached = true;
351  }
352  }
353  else
354  {
355  // Get out of infinite loop (for instance due to low accuracy state functions,
356  // to stringent tolerance or limit case for trop. corrections).
357  if( counter == 50 )
358  {
359  isToleranceReached = true;
360  std::string errorMessage =
361  "Warning, light time unconverged at level " +
362  boost::lexical_cast< std::string >(
363  std::fabs( newLightTimeCalculation - previousLightTimeCalculation ) ) +
364  "; current light-time corrections are: " +
365  std::to_string( currentCorrection_ ) + " and input time was " +
366  std::to_string( static_cast< double >( time ) );
367  std::cerr << errorMessage << std::endl;
368  }
369 
370  // Update light time for new iteration.
371  previousLightTimeCalculation = newLightTimeCalculation;
372  }
373 
374  counter++;
375  }
376 
377  // Set output variables and return the light time.
378  receiverStateOutput = receiverState;
379  transmitterStateOutput = transmitterState;
380 
381  return newLightTimeCalculation;
382  }

393  Eigen::Matrix< ObservationScalarType, 1, 3 > getPartialOfLightTimeWrtLinkEndPosition(
394  const StateType& transmitterState,
395  const StateType& receiverState,
396  const TimeType transmitterTime,
397  const TimeType receiverTime,
398  const bool isPartialWrtReceiver )
399  {
400  setTotalLightTimeCorrection( transmitterState, receiverState, transmitterTime, receiverTime );
401 
402  Eigen::Matrix< ObservationScalarType, 3, 1 > relativePosition =
403  receiverState.segment( 0, 3 ) - transmitterState.segment( 0, 3 );
404  return ( relativePosition.normalized( ) ).transpose( ) *
405  ( mathematical_constants::getFloatingInteger< ObservationScalarType >( 1 ) +
406  currentCorrection_ / relativePosition.norm( ) ) *
