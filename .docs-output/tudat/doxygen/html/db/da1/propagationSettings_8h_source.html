289 template< typename StateScalarType = double >
290 Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > getConcatenatedInitialStates(
291  const std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > >& singleArcPropagatorSettings )
292 {
293  // Define size of return vector
294  int vectorSize = getConcatenatedStateSize( singleArcPropagatorSettings );
295  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialStates =
296  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >( vectorSize );
297 
298  // Retrieve single-arc initial states arc-by-arac
299  int currentIndex = 0;
300  int currentBlockSize = 0;
301  for( unsigned int i = 0; i < singleArcPropagatorSettings.size( ); i++ )
302  {
303  currentBlockSize = singleArcPropagatorSettings.at( i )->getConventionalStateSize( );
304  initialStates.segment( currentIndex, currentBlockSize ) = singleArcPropagatorSettings.at( i )->getInitialStates( );
305  currentIndex += currentBlockSize;
306  }
307 
308  return initialStates;
309 }
