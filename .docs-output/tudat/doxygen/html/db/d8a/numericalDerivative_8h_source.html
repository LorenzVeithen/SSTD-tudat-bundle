71 template< typename InputType, typename ResultType >
72 ResultType computeCentralDifference( const InputType& input, const int derivativeIndex,
73  const std::function< ResultType( const InputType& ) >& function,
74  double minimumStep = 0.0, double relativeStepSize = 0.0,
75  CentralDifferenceOrders order = order2 )
76 {
77  const std::map< int, double >& coefficients = getCentralDifferenceCoefficients( order );
78 
79  if ( minimumStep == 0.0 )
80  {
81  // Set the minimum step to a fourth of the amount of significant
82  // digits in a double precision floating point.
83  minimumStep = std::pow( 2.0, -13 );
84  }
85 
86  if ( relativeStepSize == 0.0 )
87  {
88  // Set the relative step to half of the amount of significant digits
89  // in a double precision floating point.
90  relativeStepSize = std::pow( 2.0, -26 );
91  }
92 
93  // Ensure proper rounding by storing the step in a temporary volatile, see
94  // (Press W.H., et al., 2002).
95  const volatile double temporaryVariable = input( derivativeIndex ) +
96  std::max( minimumStep, std::abs( relativeStepSize * input( derivativeIndex ) ) );
97  const double realStepSize = temporaryVariable - input( derivativeIndex );
98 
99  ResultType result;
100 
101  // Compute the numerical derivative.
102  for ( std::map< int, double >::const_iterator coefficient = coefficients.begin( );
103  coefficient != coefficients.end( ); coefficient++ )
104  {
105  // Generate deviated input.
106  InputType deviatedInput( input );
107  deviatedInput( derivativeIndex ) += coefficient->first * realStepSize;
108 
109  // Evaluate the function.
110  ResultType deviatedResult = function( deviatedInput );
111 
112  // Store the result.
113  if ( result.size( ) == 0 )
114  {
115  // Initialize the result.
116  result = ResultType::Zero( deviatedResult.rows( ), deviatedResult.cols( ) );
117  }
118 
119  result += deviatedResult * ( coefficient->second / realStepSize );
120  }
121 
122  return result;
123 }

143 template< typename DependentVariableType, typename IndependentVariableType >
144 DependentVariableType computeCentralDifferenceFromFunction(
145  const std::function< DependentVariableType( const IndependentVariableType ) >& dependentVariableFunction,
146  const IndependentVariableType nominalIndependentVariable,
147  const IndependentVariableType independentVariableStepSize,
148  CentralDifferenceOrders order = order2 )
149 {
150  const std::map< int, double >& coefficients = getCentralDifferenceCoefficients( order );
151 
152  IndependentVariableType perturbedInput;
153  DependentVariableType perturbedOutput;
154  DependentVariableType numericalDerivative;
155 
156  // Compute the numerical derivative.
157  for ( std::map< int, double >::const_iterator coefficientIterator = coefficients.begin( );
158  coefficientIterator != coefficients.end( ); coefficientIterator++ )
159  {
160  // Generate deviated input.
161  perturbedInput = nominalIndependentVariable + coefficientIterator->first * independentVariableStepSize;
162  perturbedOutput = dependentVariableFunction( perturbedInput );
163 
164  if( coefficientIterator == coefficients.begin( ) )
165  {
166  numericalDerivative = perturbedOutput * ( coefficientIterator->second / independentVariableStepSize );
167  }
168  else
169  {
170  numericalDerivative += perturbedOutput * ( coefficientIterator->second / independentVariableStepSize );
171  }
172  }
173 
174  return numericalDerivative;
