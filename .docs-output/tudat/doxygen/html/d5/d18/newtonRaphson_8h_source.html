97  DataType execute( const FunctionPointer rootFunction, const DataType initialGuess )
98  {
99  // Set the root function.
100  this->rootFunction = rootFunction;
101 
102  // Start at initial guess, and compute the function value and its first derivative.
103  DataType currentRootValue = TUDAT_NAN;
104  DataType nextRootValue = initialGuess;
105  DataType currentFunctionValue = TUDAT_NAN;
106  DataType nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
107  DataType currentDerivativeValue = TUDAT_NAN;
108  DataType nextDerivativeValue = this->rootFunction->computeDerivative( 1, nextRootValue );
109 
110  if( !( nextFunctionValue == mathematical_constants::getFloatingInteger< DataType >( 0 ) ) )
111  {
112  // Loop counter.
113  unsigned int counter = 1;
114 
115  // Loop until we have a solution with sufficient accuracy.
116  do
117  {
118  // Save the old values.
119  currentRootValue = nextRootValue;
120  currentFunctionValue = nextFunctionValue;
121  currentDerivativeValue = nextDerivativeValue;
122 
123  // Compute next value of root using the following algorithm (see class documentation):
124  nextRootValue = currentRootValue -
125  currentFunctionValue / currentDerivativeValue;
126  nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
127  nextDerivativeValue = this->rootFunction->computeDerivative( 1, nextRootValue );
128 
129  // Update the counter.
130  counter++;
131  }
132  while( nextFunctionValue != mathematical_constants::getFloatingInteger< DataType >( 0 ) &&
133  !this->terminationFunction_( nextRootValue, currentRootValue, nextFunctionValue,
134  currentFunctionValue, counter ) );
135  }
136 
137  return nextRootValue;
138  }
