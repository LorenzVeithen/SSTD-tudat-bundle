216  template< typename TimeType >
217  std::pair< Eigen::Vector5d, TimeType > getRotationAnglesFromItrsToGcrs(
218  const double timeValue,
219  basic_astrodynamics::TimeScales timeScale = basic_astrodynamics::tt_scale )
220  {
221  // Compute required time values
222  TimeType terrestrialTime = terrestrialTimeScaleConverter_->getCurrentTime< TimeType >(
223  timeScale, basic_astrodynamics::tt_scale, timeValue, Eigen::Vector3d::Zero( ) );
224  TimeType utc = terrestrialTimeScaleConverter_->getCurrentTime< TimeType >(
225  timeScale, basic_astrodynamics::utc_scale, timeValue, Eigen::Vector3d::Zero( ) );
226  TimeType ut1 = terrestrialTimeScaleConverter_->getCurrentTime< TimeType >(
227  timeScale, basic_astrodynamics::ut1_scale, timeValue, Eigen::Vector3d::Zero( ) );
228 
229  // Compute nutation/precession parameters
230  std::pair< Eigen::Vector2d, double > positionOfCipInGcrs =
231  precessionNutationCalculator_->getPositionOfCipInGcrs(
232  terrestrialTime, utc );
233 
234  // Compute polar motion values
235  Eigen::Vector2d positionOfCipInItrs = polarMotionCalculator_->getPositionOfCipInItrs(
236  terrestrialTime, utc );
237 
238  // Return vector of angles.
239  Eigen::Vector5d rotationAngles;
240  rotationAngles[ 0 ] = positionOfCipInGcrs.first.x( );
241  rotationAngles[ 1 ] = positionOfCipInGcrs.first.y( );
242  rotationAngles[ 2 ] = positionOfCipInGcrs.second;
243  rotationAngles[ 3 ] = positionOfCipInItrs.x( );
244  rotationAngles[ 4 ] = positionOfCipInItrs.y( );
245  return std::make_pair( rotationAngles, ut1 );
246  }

314 template< typename UT1ScalarType >
315 std::pair< std::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::Matrix< double, 5, 1 > > >,
316 std::shared_ptr< interpolators::OneDimensionalInterpolator< double, UT1ScalarType > > >
317 createInterpolatorsForItrsToGcrsAngles(
318  const double intervalStart, const double intervalEnd, const double timeStep,
319  const basic_astrodynamics::TimeScales timeScale = basic_astrodynamics::tdb_scale,
320  const std::shared_ptr< EarthOrientationAnglesCalculator > earthOrientationCalculator =
321  createStandardEarthOrientationCalculator( ),
322  const std::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
323  std::make_shared< interpolators::LagrangeInterpolatorSettings >( 6 ) )
324 {
325  // Define interpolators
326  std::map< double, Eigen::Matrix< double, 5, 1 > > anglesMap;
327  std::map< double, UT1ScalarType > ut1Map;
328 
329  // Iterate over all times and compute rotation parameters
330  std::pair< Eigen::Vector5d, UT1ScalarType > currentRotationValues;
331  double currentTime = intervalStart;
332  while( currentTime < intervalEnd )
333  {
334  currentRotationValues = earthOrientationCalculator->getRotationAnglesFromItrsToGcrs< UT1ScalarType >(
335  currentTime, timeScale );
336  anglesMap[ currentTime ] = currentRotationValues.first;
337  ut1Map[ currentTime ] = currentRotationValues.second;
338  currentTime += timeStep;
339  }
340 
341  // Create interpolator for angles
342  std::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::Matrix< double, 5, 1 > > > anglesInterpolator =
343  interpolators::createOneDimensionalInterpolator( anglesMap, interpolatorSettings );
344 
345  // Update UT1 interpolation time step scalar
346  if( sizeof( UT1ScalarType ) == 8 )
347  {
348  interpolatorSettings->resetUseLongDoubleTimeStep( true );
349  }
350 
351  // Create interpolator for UT1
352  std::shared_ptr< interpolators::OneDimensionalInterpolator< double, UT1ScalarType > > ut1Interpolator =
353  interpolators::createOneDimensionalInterpolator( ut1Map, interpolatorSettings );
354 
355  return std::make_pair( anglesInterpolator, ut1Interpolator );
356 
357 }
