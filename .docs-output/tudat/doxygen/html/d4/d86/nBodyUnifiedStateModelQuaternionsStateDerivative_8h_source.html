141  void calculateSystemStateDerivative(
142  const TimeType time, const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
143  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
144  {
145  // Get total inertial accelerations acting on bodies
146  Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > currentAccelerationInIntertialFrame;
147  currentAccelerationInIntertialFrame.resizeLike( currentCartesianLocalSolution_ );
148  this->sumStateDerivativeContributions( stateOfSystemToBeIntegrated, currentAccelerationInIntertialFrame, false );
149 
150  // Compute RSW accelerations for each body, and evaluate equations of motion for USM7 elements.
151  stateDerivative.setZero( );
152  Eigen::Vector3d currentAccelerationInRswFrame;
153  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
154  {
155  currentAccelerationInRswFrame = reference_frames::getInertialToRswSatelliteCenteredFrameRotationMatrix(
156  currentCartesianLocalSolution_.segment( i * 6, 6 ).template cast< double >( ) ) *
157  currentAccelerationInIntertialFrame.block( i * 6 + 3, 0, 3, 1 ).template cast< double >( );
158 
159  stateDerivative.block( i * 7, 0, 7, 1 ) = computeStateDerivativeForUnifiedStateModelQuaternions(
160  stateOfSystemToBeIntegrated.block( i * 7, 0, 7, 1 ).template cast< double >( ), currentAccelerationInRswFrame,
161  centralBodyGravitationalParameters_.at( i )( ) ).template cast< StateScalarType >( );
162  }

246  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
247  {
248  // Loop over each body
249  const double tolerance = 20.0 * std::numeric_limits< double >::epsilon( );
250  Eigen::Matrix< StateScalarType, 4, 1 > quaternionsVector;
251  StateScalarType quaternionsMagnitude;
252  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
253  {
254  // Normalize quaternions
255  quaternionsVector = unprocessedState.block( i * 7 + 3, 0, 4, 1 );
256  quaternionsMagnitude = quaternionsVector.norm( );
257  if ( std::fabs( quaternionsMagnitude - 1.0 ) >= tolerance )
258  {
259  // Normalize
260  quaternionsVector /= quaternionsMagnitude;
261 
262  // Replace old quaternions with normalized quaternions
263  unprocessedState.segment( i * 7 + 3, 4 ) = quaternionsVector;
264  }
265  }
266  }
