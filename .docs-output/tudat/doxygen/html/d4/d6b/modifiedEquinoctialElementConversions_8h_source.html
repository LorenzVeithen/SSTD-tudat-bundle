63 template< typename ScalarType = double >
64 Eigen::Matrix< ScalarType, 6, 1 > convertKeplerianToModifiedEquinoctialElements(
65  const Eigen::Matrix< ScalarType, 6, 1 >& keplerianElements,
66  const bool flipSingularityToZeroInclination )
67 {
68  using mathematical_constants::getPi;
69  using mathematical_constants::getFloatingInteger;
70 
71  // Declaring eventual output vector.
72  Eigen::Matrix< ScalarType, 6, 1 > modifiedEquinoctialState( 6 );
73 
74  // Compute semi-latus rectum.
75  ScalarType singularityTolerance = 5.0 * std::numeric_limits< ScalarType >::epsilon( );
76 
77  // Extracting eccentricity for ease of referencing.
78  ScalarType eccentricity = keplerianElements( eccentricityIndex );
79 
80  // If e is (very near) one, then semi-major axis is undefined and thus the first kepler
81  // element is the semi-latus rectum.
82  if ( std::fabs( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) < singularityTolerance )
83  {
84  modifiedEquinoctialState( semiLatusRectumIndex )
85  = keplerianElements( semiLatusRectumIndex );
86  }
87  else // eccentricity is significantly away from singularity and can be computed with p=a(1-e).
88  {
89  modifiedEquinoctialState( semiLatusRectumIndex ) =
90  keplerianElements( semiMajorAxisIndex )
91  * ( getFloatingInteger< ScalarType >( 1 ) - eccentricity * eccentricity );
92  }
93 
94  // Determine prograde-ness, as other five parameters depend on that fact.
95  ScalarType inclination = keplerianElements( inclinationIndex );
96 
97  // If inclination is outside range [0,PI].
98  if ( ( inclination < getFloatingInteger< ScalarType >( 0 ) ) || ( inclination > getPi< ScalarType >( ) ) )
99  {
100  // Define the error message.
101  throw std::runtime_error( "Inclination is expected in range [0," + std::to_string( getPi< ScalarType >( ) ) + "]\n"
102  + "Specified inclination: " + std::to_string( inclination ) + " rad." );
103  }
104  //Else, nothing wrong and continue.
105 
106  // Compute set dependant helper parameters.
107  ScalarType argumentOfPeriapsisAndAscendingNode = getFloatingInteger< ScalarType >( 0 );
108  ScalarType tangentOfHalfInclination = getFloatingInteger< ScalarType >( 0 );
109 
110  // If normal set of equations is to be used (i.e. not inverse set for singularity).
111  if ( !flipSingularityToZeroInclination )
112  {
113  // Add (+) argument of periapsis and longitude of ascending node.
114  argumentOfPeriapsisAndAscendingNode = keplerianElements( argumentOfPeriapsisIndex )
115  + keplerianElements( longitudeOfAscendingNodeIndex );
116 
117  // Take the tangent of half the inclination.
118  tangentOfHalfInclination = std::tan( inclination / getFloatingInteger< ScalarType >( 2 ) );
119  }
120  else // the orbit is retrograde.
121  {
122  // Subtract (-) longitude of ascending node from argument of periapsis.
123  argumentOfPeriapsisAndAscendingNode = keplerianElements( argumentOfPeriapsisIndex )
124  - keplerianElements( longitudeOfAscendingNodeIndex );
125 
126  // Take the inverse of the tangent of half the inclination to avoid singularity at tan
127  // PI/2.
128  tangentOfHalfInclination = getFloatingInteger< ScalarType >( 1 ) /
129  std::tan( inclination / getFloatingInteger< ScalarType >( 2 ) );
130  }
131 
132  // Compute f-element.
133  modifiedEquinoctialState( fElementIndex ) = eccentricity
134  * std::cos( argumentOfPeriapsisAndAscendingNode );
135 
136  // Compute g-element.
137  modifiedEquinoctialState( gElementIndex ) = eccentricity
138  * std::sin( argumentOfPeriapsisAndAscendingNode );
139 
140  // Compute h-element.
141  modifiedEquinoctialState( hElementIndex ) = tangentOfHalfInclination
142  * std::cos( keplerianElements( longitudeOfAscendingNodeIndex ) );
143 
144  // Compute k-element.
145  modifiedEquinoctialState( kElementIndex ) = tangentOfHalfInclination
146  * std::sin( keplerianElements( longitudeOfAscendingNodeIndex ) );
147 
148  // Compute true longitude (modulo 2 PI to keep within interval -2PI to 2PI).
149  modifiedEquinoctialState( trueLongitudeIndex )
150  = basic_mathematics::computeModulo(
151  argumentOfPeriapsisAndAscendingNode
152  + keplerianElements( trueAnomalyIndex ), getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
153 
154  // Give back result.
155  return modifiedEquinoctialState;
156 }

178 template< typename ScalarType = double >
179 Eigen::Matrix< ScalarType, 6, 1 > convertKeplerianToModifiedEquinoctialElements(
180  const Eigen::Matrix< ScalarType, 6, 1 >& keplerianElements )
181 // Based on Hintz, 2008.
182 {
183  // Check if orbit is retrograde
184  bool flipSingularityToZeroInclination =
185  mission_geometry::isOrbitRetrograde( keplerianElements );
186 
187  // Convert to modified equinoctial elements.
188  return convertKeplerianToModifiedEquinoctialElements( keplerianElements,
189  flipSingularityToZeroInclination );
190 }

218 template< typename ScalarType = double >
219 Eigen::Matrix< ScalarType, 6, 1 > convertModifiedEquinoctialToKeplerianElements(
220  const Eigen::Matrix< ScalarType, 6, 1 >& modifiedEquinoctialElements,
221  const bool flipSingularityToZeroInclination )
222 {
223  using mathematical_constants::getPi;
224  using mathematical_constants::getFloatingInteger;
225 
226  // Declaration of output vector.
227  Eigen::Matrix< ScalarType, 6, 1 > convertedKeplerianElements = Eigen::Matrix< ScalarType, 6, 1 >::Zero( 6 );
228 
229  // for ease of referencing, almost all modified equinoctial elements.
230  ScalarType fElement = modifiedEquinoctialElements( fElementIndex );
231  ScalarType gElement = modifiedEquinoctialElements( gElementIndex );
232  ScalarType hElement = modifiedEquinoctialElements( hElementIndex );
233  ScalarType kElement = modifiedEquinoctialElements( kElementIndex );
234 
235  // Tolerance for singularities.
236  ScalarType singularityTolerance = 5.0 * std::numeric_limits< ScalarType >::epsilon( );
237 
238  // Compute eccentricity.0
239  ScalarType eccentricity = std::sqrt( fElement * fElement + gElement * gElement );
240  convertedKeplerianElements( eccentricityIndex ) = eccentricity;
241 
242  // Compute semi-major axis.
243  // If eccentricity is not near-parabolic.
244  if ( std::fabs( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) > singularityTolerance )
245  {
246  // Use semi-latus rectum and eccentricity to calculate semi-major axis with a=p/(1-e^2).
247  convertedKeplerianElements( semiMajorAxisIndex ) =
248  modifiedEquinoctialElements( semiLatusRectumIndex ) /
249  ( getFloatingInteger< ScalarType >( 1 ) - eccentricity * eccentricity );
250  }
251  // This is (almost) a parabola and the semimajor axis will tend to infinity and be useless.
252  else
253  {
254  // Give semi-latus rectum instead.
255  convertedKeplerianElements( semiLatusRectumIndex ) =
256  modifiedEquinoctialElements( semiLatusRectumIndex );
257  }
258 
259  // Compute longitude of ascending node.
260 
261  // Compute solution in [-PI,PI] interval with atan2.
262  ScalarType longitudeOfAscendingNode = std::atan2( kElement, hElement );
263 
264  // Store longitude of ascending node.
265  convertedKeplerianElements( longitudeOfAscendingNodeIndex ) = basic_mathematics::
266  computeModulo( longitudeOfAscendingNode, getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
267 
268  // Compute inclination.
269 
270  // If prograde factor must be positive, otherwise negative.
271  ScalarType retrogradeFactor =
272  flipSingularityToZeroInclination ? -getFloatingInteger< ScalarType >( 1 ) : getFloatingInteger< ScalarType >( 1 );
273 
274  // Calculate magnitude of inclination with retrogradeFactor (derived based on Hintz, 2008).
275  ScalarType hSquaredPlusKSquared = std::pow( hElement * hElement + kElement * kElement ,
276  retrogradeFactor );
277 
278  convertedKeplerianElements( inclinationIndex ) =
279  getFloatingInteger< ScalarType >( 2 ) * std::atan( std::sqrt( hSquaredPlusKSquared ) );
280  // Was: std::atan2(2 * std::sqrt(hSquaredPlusKSquared) , (1.0 - hSquaredPlusKSquared));
281 
282  // Compute argument of periapsis.
283 
284  // Helper quantity (omega + I * OMEGA ), for argument of periapsis and true anomaly.
285  ScalarType argumentOfPeriapsisAndLongitude;
286 
287  // If eccentricity is (near) circular
288  if ( eccentricity < singularityTolerance )
289  // Then the composite argument of periapsis and longitude together cannot be determined
290  // because both arguments of the atan2 functions will be zero.
291  {
292  // Set argument of periapsis to zero.
293  convertedKeplerianElements( argumentOfPeriapsisIndex ) = getFloatingInteger< ScalarType >( 0 );
294 
295  // Set composite argument to longitude only, since argument of periapsis is now zero.
296  argumentOfPeriapsisAndLongitude = retrogradeFactor * longitudeOfAscendingNode;
297  }
298  else
299  // Argument of periapsis can be found from composite argument.
300  {
301  // Compute composite argument.
302  argumentOfPeriapsisAndLongitude = std::atan2( gElement, fElement );
303 
304  // Compute argument of periapsis.
305  convertedKeplerianElements( argumentOfPeriapsisIndex ) = basic_mathematics::
306  computeModulo( argumentOfPeriapsisAndLongitude
307  - retrogradeFactor * longitudeOfAscendingNode,
308  getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
309  }
310 
311  // Compute true anomaly.
312  convertedKeplerianElements( trueAnomalyIndex ) = basic_mathematics::
313  computeModulo( modifiedEquinoctialElements( trueLongitudeIndex )
314  - argumentOfPeriapsisAndLongitude, getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
315 
316  // Return converted elements.
317  return convertedKeplerianElements;
318 }

433 template< typename ScalarType = double >
434 Eigen::Matrix< ScalarType, 6, 1 > convertModifiedEquinoctialToCartesianElementsViaKeplerElements(
435  const Eigen::Matrix< ScalarType, 6, 1 >& modifiedEquinoctialElements,
436  const ScalarType centralBodyGravitationalParameter,
437  const bool flipSingularityToZeroInclination )
438 // Using unnamed pdf and code archive Bart Rmgens.
439 {
440  using mathematical_constants::getFloatingInteger;
441 
442  // Creating output vector.
443  Eigen::Matrix< ScalarType, 6, 1 > convertedCartesianElements = Eigen::Matrix< ScalarType, 6, 1 >::Zero( 6 );
444 
445  // If the prograde equations are to be used.
446  if ( !flipSingularityToZeroInclination )
447  {
448  // Local storage of elements for ease of access.
449  ScalarType semiLatusRectum = modifiedEquinoctialElements( semiLatusRectumIndex );
450  ScalarType fElement = modifiedEquinoctialElements( fElementIndex );
451  ScalarType gElement = modifiedEquinoctialElements( gElementIndex );
452  ScalarType hElement = modifiedEquinoctialElements( hElementIndex );
453  ScalarType kElement = modifiedEquinoctialElements( kElementIndex );
454  ScalarType trueLongitude = modifiedEquinoctialElements( trueLongitudeIndex );
455 
456  // Computing helper parameters.
457 
458  // Square-root
459  ScalarType squareRootOfGravitationalParameterOverSemiLatusRectum =
460  std::sqrt( centralBodyGravitationalParameter / semiLatusRectum );
461 
462  // Cosines and sine.
463  ScalarType cosineTrueLongitude = std::cos( trueLongitude );
464  ScalarType sineTrueLongitude = std::sin( trueLongitude );
465 
466  // s, a, w, r from references.
467  ScalarType sSquaredParameter = getFloatingInteger< ScalarType >( 1 ) + hElement * hElement + kElement * kElement;
468  ScalarType aSquaredParameter = hElement * hElement - kElement*kElement;
469  ScalarType wParameter = getFloatingInteger< ScalarType >( 1 ) + fElement * cosineTrueLongitude
470  + gElement * sineTrueLongitude;
471  ScalarType radius = semiLatusRectum / wParameter;
472 
473  // Computing position and storing (using code archive Bart Rmgens and unnamed pdf).
474  convertedCartesianElements( xCartesianPositionIndex )
475  = radius / sSquaredParameter *
476  ( cosineTrueLongitude + aSquaredParameter
477  * cosineTrueLongitude + getFloatingInteger< ScalarType >( 2 ) * hElement * kElement
478  * sineTrueLongitude );
479  convertedCartesianElements( yCartesianPositionIndex )
480  = radius / sSquaredParameter *
481  ( sineTrueLongitude - aSquaredParameter
482  * sineTrueLongitude + getFloatingInteger< ScalarType >( 2 ) * hElement * kElement
483  * cosineTrueLongitude );
484  convertedCartesianElements( zCartesianPositionIndex ) = getFloatingInteger< ScalarType >( 2 ) * radius / sSquaredParameter
485  * ( hElement * sineTrueLongitude - kElement * cosineTrueLongitude );
486 
487  // Computing velocities (these can probably use more optimal computation).
488  convertedCartesianElements( xCartesianVelocityIndex ) = -getFloatingInteger< ScalarType >( 1 ) / sSquaredParameter
489  * squareRootOfGravitationalParameterOverSemiLatusRectum
490  * ( sineTrueLongitude + aSquaredParameter * sineTrueLongitude
491  - getFloatingInteger< ScalarType >( 2 ) * hElement * kElement * cosineTrueLongitude + gElement
492  - getFloatingInteger< ScalarType >( 2 ) * fElement * hElement * kElement + aSquaredParameter * gElement );
493  convertedCartesianElements( yCartesianVelocityIndex ) = -getFloatingInteger< ScalarType >( 1 ) / sSquaredParameter
494  * squareRootOfGravitationalParameterOverSemiLatusRectum
495  * ( -cosineTrueLongitude + aSquaredParameter * cosineTrueLongitude
496  + getFloatingInteger< ScalarType >( 2 ) * hElement * kElement * sineTrueLongitude - fElement
497  + getFloatingInteger< ScalarType >( 2 ) * gElement * hElement * kElement + aSquaredParameter * fElement );
498  convertedCartesianElements( zCartesianVelocityIndex ) = getFloatingInteger< ScalarType >( 2 ) / sSquaredParameter
499  * squareRootOfGravitationalParameterOverSemiLatusRectum
500  * ( hElement * cosineTrueLongitude + kElement * sineTrueLongitude
501  + fElement * hElement + gElement * kElement );
502  }
503  else // use the indirect transformation via an intermediate Kepler state.
504  {
505  // Use the Keplerian state as an intermediary step, as there is no direct transformation
506  // available in literature (personal derivation forgone for now due to time constraints).
507  convertedCartesianElements =
508  convertKeplerianToCartesianElements(
509  convertModifiedEquinoctialToKeplerianElements(
510  modifiedEquinoctialElements, flipSingularityToZeroInclination ),
511  centralBodyGravitationalParameter );
512  }
513 
514  // Return converted set of elements.
515  return convertedCartesianElements;
516 }

544 template< typename ScalarType = double >
545 Eigen::Matrix< ScalarType, 6, 1 > convertModifiedEquinoctialToCartesianElements(
546  const Eigen::Matrix< ScalarType, 6, 1 >& modifiedEquinoctialElements,
547  const ScalarType centralBodyGravitationalParameter,
548  const bool flipSingularityToZeroInclination )
549 {
550  if( flipSingularityToZeroInclination )
551  {
552  throw std::runtime_error(
553  "Error in direct conversion from MEE to Cartesian elements, implementation with singularity at 0 degrees not yet implemented" );
554  }
555  using mathematical_constants::getFloatingInteger;
556 
557  // Retrieve MEE components and compute intermediate quantities
558  ScalarType semiLatusRectrum = modifiedEquinoctialElements( semiParameterIndex );
559  ScalarType angularMomentumPerUnitGravitationalParameter =
560  std::sqrt( semiLatusRectrum / centralBodyGravitationalParameter );
561 
562  ScalarType sineTrueLongitude = std::sin( modifiedEquinoctialElements( trueLongitudeIndex ) );
563  ScalarType cosineTrueLongitude = std::cos( modifiedEquinoctialElements( trueLongitudeIndex ) );
564 
565  ScalarType parameterF = modifiedEquinoctialElements( fElementIndex );
566  ScalarType parameterG = modifiedEquinoctialElements( gElementIndex );
567  ScalarType parameterH = modifiedEquinoctialElements( hElementIndex );
568  ScalarType parameterK = modifiedEquinoctialElements( kElementIndex );
569 
570  ScalarType parameterW =
571  getFloatingInteger< ScalarType >( 1 ) + parameterF * cosineTrueLongitude + parameterG * sineTrueLongitude;
572  ScalarType parameterSSquared = getFloatingInteger< ScalarType >( 1 ) + parameterH * parameterH + parameterK * parameterK;
573  ScalarType parameterAlphaSquared = parameterH * parameterH - parameterK * parameterK;
574 
575  // Compute Cartesian elements
576  Eigen::Matrix< ScalarType, 6, 1 > cartesianElements;
577  cartesianElements( 0 ) = cosineTrueLongitude + parameterAlphaSquared * cosineTrueLongitude +
578  getFloatingInteger< ScalarType >( 2 ) * parameterH * parameterK * sineTrueLongitude;
579  cartesianElements( 1 ) = ( sineTrueLongitude - parameterAlphaSquared * sineTrueLongitude +
580  getFloatingInteger< ScalarType >( 2 ) * parameterH * parameterK * cosineTrueLongitude );
581  cartesianElements( 2 ) =
582  getFloatingInteger< ScalarType >( 2 ) * ( parameterH * sineTrueLongitude - parameterK * cosineTrueLongitude );
583  cartesianElements.segment( 0, 3 ) *= semiLatusRectrum / ( parameterW * parameterSSquared );
584 
585  cartesianElements( 3 ) =
586  -( sineTrueLongitude + parameterAlphaSquared * sineTrueLongitude -
587  getFloatingInteger< ScalarType >( 2 ) * parameterH * parameterK * cosineTrueLongitude + parameterG -
588  getFloatingInteger< ScalarType >( 2 ) * parameterF * parameterH * parameterK +
589  parameterAlphaSquared * parameterG );
590  cartesianElements( 4 ) = -( -cosineTrueLongitude + parameterAlphaSquared * cosineTrueLongitude +
591  getFloatingInteger< ScalarType >( 2 ) * parameterH * parameterK * sineTrueLongitude -
592  parameterF + getFloatingInteger< ScalarType >( 2 ) * parameterG * parameterH * parameterK +
593  parameterAlphaSquared * parameterF );
594  cartesianElements( 5 ) =
595  getFloatingInteger< ScalarType >( 2 ) * ( parameterH * cosineTrueLongitude + parameterK * sineTrueLongitude +
596  parameterF * parameterH + parameterG * parameterK );
597  cartesianElements.segment( 3, 3 ) *=
598  getFloatingInteger< ScalarType >( 1 ) / ( parameterSSquared * angularMomentumPerUnitGravitationalParameter );
599 
600  return cartesianElements;
