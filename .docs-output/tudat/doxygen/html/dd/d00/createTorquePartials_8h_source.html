56 template< typename InitialStateParameterType = double >
57 std::shared_ptr< acceleration_partials::TorquePartial > createAnalyticalTorquePartial(
58  std::shared_ptr< basic_astrodynamics::TorqueModel > torqueModel,
59  const std::pair< std::string, std::shared_ptr< simulation_setup::Body > > acceleratedBody,
60  const std::pair< std::string, std::shared_ptr< simulation_setup::Body > > acceleratingBody,
61  const basic_astrodynamics::SingleBodyTorqueModelMap& torqueVector = basic_astrodynamics::SingleBodyTorqueModelMap( ),
62  const simulation_setup::SystemOfBodies& bodies = simulation_setup::SystemOfBodies( ),
63  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< InitialStateParameterType > >
64  parametersToEstimate =
65  std::shared_ptr< estimatable_parameters::EstimatableParameterSet< InitialStateParameterType > >( ) )
66 {
67  using namespace gravitation;
68  using namespace basic_astrodynamics;
69  using namespace electromagnetism;
70  using namespace aerodynamics;
71  using namespace acceleration_partials;
72 
73  std::shared_ptr< acceleration_partials::TorquePartial > torquePartial;
74 
75  // Identify current torque model type
76  AvailableTorque torqueType = getTorqueModelType( torqueModel );
77  switch( torqueType )
78  {
79  case second_order_gravitational_torque:
80 
81  // Check if identifier is consistent with type.
82  if( std::dynamic_pointer_cast< SecondDegreeGravitationalTorqueModel >( torqueModel ) == nullptr )
83  {
84  throw std::runtime_error( "Torque class type does not match torque type (second_order_gravitational_torque) when making torque partial" );
85  }
86  else
87  {
88  // Create partial-calculating object.
89  std::function< double( ) > inertiaTensorNormalizationFunction;
90  if( std::dynamic_pointer_cast< SphericalHarmonicsGravityField >( acceleratedBody.second->getGravityFieldModel( ) )
91  != nullptr )
92  {
93  inertiaTensorNormalizationFunction =
94  std::bind( &SphericalHarmonicsGravityField::getInertiaTensorNormalizationFactor,
95  std::dynamic_pointer_cast< SphericalHarmonicsGravityField >(
96  acceleratedBody.second->getGravityFieldModel( ) ) );
97  }
98  torquePartial = std::make_shared< SecondDegreeGravitationalTorquePartial >
99  ( std::dynamic_pointer_cast< SecondDegreeGravitationalTorqueModel >( torqueModel ),
100  inertiaTensorNormalizationFunction, acceleratedBody.first, acceleratingBody.first );
101  }
102  break;
103  case spherical_harmonic_gravitational_torque:
104 
105  // Check if identifier is consistent with type.
106  if( std::dynamic_pointer_cast< SphericalHarmonicGravitationalTorqueModel >( torqueModel ) == nullptr )
107  {
108  throw std::runtime_error( "Torque class type does not match torque type (spherical_harmonic_gravitational_torque) when making torque partial" );
109  }
110  else
111  {
112  // Create partial-calculating object.
113  torquePartial = std::make_shared< SphericalHarmonicGravitationalTorquePartial >
114  ( std::dynamic_pointer_cast< SphericalHarmonicGravitationalTorqueModel >( torqueModel ),
115  std::dynamic_pointer_cast< SphericalHarmonicsGravityPartial >( createAnalyticalAccelerationPartial(
116  std::dynamic_pointer_cast< SphericalHarmonicGravitationalTorqueModel >(
117  torqueModel )->getSphericalHarmonicAcceleration( ), acceleratingBody, acceleratedBody,
118  bodies, parametersToEstimate ) ),
119  acceleratedBody.first, acceleratingBody.first );
120  }
121  break;
122  case inertial_torque:
123  {
124  std::function< Eigen::Vector3d( ) > angularVelocityFunction =
125  std::bind( &Body::getCurrentAngularVelocityVectorInLocalFrame, acceleratedBody.second );
126  std::function< Eigen::Matrix3d( ) > inertiaTensorFunction =
127  std::bind( &Body::getBodyInertiaTensor, acceleratedBody.second );
128 
129  std::function< double( ) > gravitationalParameterFunction;
130  if( acceleratedBody.second->getGravityFieldModel( ) != nullptr )
131  {
132  gravitationalParameterFunction =
133  std::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
134  acceleratedBody.second->getGravityFieldModel( ) );
135  }
136 
137  std::function< double( ) > inertiaTensorNormalizationFunction;
138  if( std::dynamic_pointer_cast< gravitation::SphericalHarmonicsGravityField >( acceleratedBody.second->getGravityFieldModel( ) )
139  != nullptr )
140  {
141  inertiaTensorNormalizationFunction =
142  std::bind( &gravitation::SphericalHarmonicsGravityField::getInertiaTensorNormalizationFactor,
143  std::dynamic_pointer_cast< gravitation::SphericalHarmonicsGravityField >(
144  acceleratedBody.second->getGravityFieldModel( ) ) );
145  }
146 
147  torquePartial = std::make_shared< acceleration_partials::InertialTorquePartial >(
148  angularVelocityFunction, inertiaTensorFunction, inertiaTensorNormalizationFunction,
149  gravitationalParameterFunction, acceleratedBody.first );
150  break;
151  }
152  case custom_torque:
153  throw std::runtime_error( "Custom torque does not yet have an associated partial." );
154  break;
155  default:
156  std::string errorMessage = "Torque model " + std::to_string( torqueType ) +
157  " not found when making torque partial";
158  throw std::runtime_error( errorMessage );
159  break;
160  }
161 
162  return torquePartial;
163 }

176 template< typename InitialStateParameterType >
177 orbit_determination::StateDerivativePartialsMap createTorquePartialsMap(
178  const basic_astrodynamics::TorqueModelMap& torqueMap,
179  const simulation_setup::SystemOfBodies& bodies,
180  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< InitialStateParameterType > >
181  parametersToEstimate )
182 {
183  // Declare return map.
184  orbit_determination::StateDerivativePartialsMap torquePartialsList;
185 
186  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
187  Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
188  estimatable_parameters::getListOfRotationalStateParametersToEstimate( parametersToEstimate );
189  torquePartialsList.resize( initialDynamicalParameters.size( ) );
190 
191  // Iterate over list of bodies of which the partials of the torques acting on them are required.
192  for( basic_astrodynamics::TorqueModelMap::const_iterator torqueIterator = torqueMap.begin( );
193  torqueIterator != torqueMap.end( ); torqueIterator++ )
194  {
195  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
196  {
197  if( initialDynamicalParameters.at( i )->getParameterName( ).second.first == torqueIterator->first )
198  {
199  if( ( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_rotational_body_state ) )
200  {
201  // Get object for body undergoing torque
202  const std::string acceleratedBody = torqueIterator->first;
203  std::shared_ptr< simulation_setup::Body > acceleratedBodyObject = bodies.at( acceleratedBody );
204 
205  // Retrieve list of torques acting on current body.
206  basic_astrodynamics::SingleBodyTorqueModelMap torqueVector =
207  torqueMap.at( acceleratedBody );
208 
209  // Declare list of torque partials of current body.
210  std::vector< std::shared_ptr< orbit_determination::StateDerivativePartial > > torquePartialVector;
211  torquePartialVector.push_back( createConstantTorqueRotationalDynamicsPartial(
212  std::make_pair( acceleratedBody, acceleratedBodyObject ),
213  torqueVector ) );
214 
215  // Iterate over all torque models and generate their partial-calculating objects.
216  for( basic_astrodynamics::SingleBodyTorqueModelMap::iterator
217  innerTorqueIterator = torqueVector.begin( );
218  innerTorqueIterator != torqueVector.end( ); innerTorqueIterator++ )
219  {
220  // Get object for body exerting torque
221  std::string acceleratingBody = innerTorqueIterator->first;
222  std::shared_ptr< simulation_setup::Body > acceleratingBodyObject;
223  if( acceleratingBody != "" )
224  {
225  acceleratingBodyObject = bodies.at( acceleratingBody );
226  }
227 
228  for( unsigned int j = 0; j < innerTorqueIterator->second.size( ); j++ )
229  {
230  // Create single partial object
231  std::shared_ptr< acceleration_partials::TorquePartial > currentTorquePartial =
232  createAnalyticalTorquePartial(
233  innerTorqueIterator->second[ j ],
234  std::make_pair( acceleratedBody, acceleratedBodyObject ),
235  std::make_pair( acceleratingBody, acceleratingBodyObject ),
236  torqueVector, bodies, parametersToEstimate );
237 
238  torquePartialVector.push_back( currentTorquePartial );
239  }
240  }
241 
242  // Add partials of current body's torques to vector.
243  torquePartialsList[ i ] = torquePartialVector;
244  }
245  }
246  }
247  }
248  return torquePartialsList;
249 }
