66 template< typename ScalarType = double >
67 ScalarType computeKeplersFunctionForEllipticalOrbits( const ScalarType eccentricAnomaly,
68  const ScalarType eccentricity,
69  const ScalarType meanAnomaly )
70 {
71  return eccentricAnomaly - eccentricity * std::sin( eccentricAnomaly ) - meanAnomaly;
72 }

108 template< typename ScalarType = double >
109 ScalarType computeKeplersFunctionForHyperbolicOrbits( const ScalarType hyperbolicEccentricAnomaly,
110  const ScalarType eccentricity,
111  const ScalarType hyperbolicMeanAnomaly )
112 {
113  return eccentricity * std::sinh( hyperbolicEccentricAnomaly )
114  - hyperbolicEccentricAnomaly - hyperbolicMeanAnomaly;

130 ScalarType computeFirstDerivativeKeplersFunctionForHyperbolicOrbits(
131  const ScalarType hyperbolicEccentricAnomaly,
132  const ScalarType eccentricity)

156 template< typename ScalarType = double >
157 ScalarType convertMeanAnomalyToEccentricAnomaly(
158  const ScalarType eccentricity, const ScalarType aMeanAnomaly,
159  const bool useDefaultInitialGuess = true,
160  const ScalarType userSpecifiedInitialGuess = TUDAT_NAN,
161  std::shared_ptr< root_finders::RootFinder< ScalarType > > rootFinder =
162  std::shared_ptr< root_finders::RootFinder< ScalarType > >( ) )
163 {
164  using namespace mathematical_constants;
165  using namespace root_finders;
166 
167 
168  // Set mean anomaly to region between 0 and 2 PI.
169  ScalarType meanAnomaly = basic_mathematics::computeModulo< ScalarType >(
170  aMeanAnomaly, getFloatingInteger< ScalarType >( 2 ) *
171  getPi< ScalarType >( ) );
172 
173  // Required because the make_shared in the function definition gives problems for MSVC.
174  if ( !rootFinder.get( ) )
175  {
176  ScalarType tolerance = 10.0 * std::numeric_limits< ScalarType >::epsilon( );
177 
178  // Loosen tolerance for near-parabolic orbits
179  if( std::fabs( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) <
180  1.0E5 * std::numeric_limits< ScalarType >::epsilon( ) )
181  {
182  tolerance *= 2.5;
183  }
184 
185  rootFinder = root_finders::createRootFinder< ScalarType >(
186  root_finders::newtonRaphsonRootFinderSettings(
187  TUDAT_NAN, tolerance, TUDAT_NAN, 20, root_finders::throw_exception ) );
188  }
189 
190  // Declare eccentric anomaly.
191  ScalarType eccentricAnomaly = TUDAT_NAN;
192 
193  // Check if orbit is elliptical, and not near-parabolic.
194  if ( eccentricity < getFloatingInteger< ScalarType >( 1 ) &&
195  eccentricity >= getFloatingInteger< ScalarType >( 0 ) )
196  {
197  // Create an object containing the function of which we whish to obtain the root from.
198  std::shared_ptr< basic_mathematics::FunctionProxy< ScalarType, ScalarType > > rootFunction
199  = std::make_shared< basic_mathematics::FunctionProxy< ScalarType, ScalarType > >(
200  std::bind( &computeKeplersFunctionForEllipticalOrbits< ScalarType >, std::placeholders::_1,
201  eccentricity, meanAnomaly ) );
202 
203  // Add the first derivative of the root function.
204  rootFunction->addBinding(
205  -1, std::bind(
206  &computeFirstDerivativeKeplersFunctionForEllipticalOrbits< ScalarType >, std::placeholders::_1,
207  eccentricity ) );
208 
209  // Declare initial guess.
210  ScalarType initialGuess = TUDAT_NAN;
211 
212  // Set the initial guess. Check if the default scheme is to be used or a user specified
213  // value should be used.
214  // !!!!!!!!!!!!! IMPORTANT !!!!!!!!!!!!!
215  // If this scheme is changed, please run a very extensive test suite. The root finder
216  // function tends to be chaotic for some very specific combinations of mean anomaly and
217  // eccentricity. Various random tests of 100.000.000 samples were done to verify the
218  // functionality of this one, and of another option for the starter: PI. [Musegaas,2012]
219  if ( useDefaultInitialGuess )
220  {
221  if ( meanAnomaly > getPi< ScalarType >( ) )
222  {
223  initialGuess = meanAnomaly - eccentricity;
224  }
225  else
226  {
227  initialGuess = meanAnomaly + eccentricity;
228  }
229  }
230  else
231  {
232  initialGuess = userSpecifiedInitialGuess;
233  }
234 
235  try
236  {
237  // Set eccentric anomaly based on result of Newton-Raphson root-finding algorithm.
238  eccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
239  }
240  // Use bisection algorithm if root finder fails
241  catch( std::runtime_error const& )
242 
243  {
244  // Set tolerance
245  ScalarType tolerance = 10.0 * std::numeric_limits< ScalarType >::epsilon( );
246 
247  // Set upper/lower bounds (loosely)
248  ScalarType lowerBound, upperBound;
249  if ( meanAnomaly > getPi< ScalarType >( ) )
250  {
251  lowerBound = 0.5 * getPi< ScalarType >( ) + 10.0 * std::numeric_limits< ScalarType >::epsilon( );
252  upperBound = mathematical_constants::getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( );
253  }
254  else
255  {
256  lowerBound = mathematical_constants::getFloatingInteger< ScalarType >( 0 );
257  upperBound = 1.5 * getPi< ScalarType >( );
258  }
259 
260  // Create root finder
261  std::shared_ptr< RootFinder< ScalarType > > bisectionRootfinder =
262  rootFinder = root_finders::createRootFinder< ScalarType >(
263  root_finders::bisectionRootFinderSettings(
264  TUDAT_NAN, TUDAT_NAN, tolerance, 100, root_finders::accept_result ),
265  lowerBound, upperBound );
266 
267  initialGuess = meanAnomaly;
268  eccentricAnomaly = bisectionRootfinder->execute( rootFunction, initialGuess );
269  }
270  }
271  // Eccentricity is invalid: eccentricity < 0.0 or eccentricity >= 1.0.
272  else
273  {
274  throw std::runtime_error( "Invalid eccentricity. Valid range is 0.0 <= e < 1.0. Eccentricity was: " +
275  std::to_string( eccentricity ) );
276  }
277 
278  // Return eccentric anomaly.
279  return eccentricAnomaly;
280 }

300 template< typename ScalarType = double >
301 ScalarType convertMeanAnomalyToHyperbolicEccentricAnomaly(
302  const ScalarType eccentricity, const ScalarType hyperbolicMeanAnomaly,
303  const bool useDefaultInitialGuess = true,
304  const ScalarType userSpecifiedInitialGuess = TUDAT_NAN,
305  std::shared_ptr< root_finders::RootFinder< ScalarType > > aRootFinder =
306  std::shared_ptr< root_finders::RootFinder< ScalarType > >( ) )
307 {
308  using namespace mathematical_constants;
309  using namespace root_finders;
310 
311 
312  std::shared_ptr< RootFinder< ScalarType > > rootFinder = aRootFinder;
313 
314  // Required because the make_shared in the function definition gives problems for MSVC.
315  if ( !rootFinder.get( ) )
316  {
317  double toleranceMultiplier = ( eccentricity - 1.0 ) < 1.0E-3 ? 10.0 : 1.0;
318  rootFinder = createRootFinder< ScalarType >(
319  newtonRaphsonRootFinderSettings(
320  TUDAT_NAN, toleranceMultiplier * 25.0 * std::numeric_limits< ScalarType >::epsilon( ),
321  TUDAT_NAN, 1000, root_finders::throw_exception ) );
322  }
323  // Declare hyperbolic eccentric anomaly.
324  ScalarType hyperbolicEccentricAnomaly = TUDAT_NAN;
325 
326  // Check if orbit is hyperbolic.
327  if ( eccentricity > getFloatingInteger< ScalarType >( 1 ) )
328  {
329  // Create an object containing the function of which we whish to obtain the root from.
330  std::shared_ptr< basic_mathematics::FunctionProxy< ScalarType, ScalarType > > rootFunction
331  = std::make_shared< basic_mathematics::FunctionProxy< ScalarType, ScalarType > >(
332  std::bind( &computeKeplersFunctionForHyperbolicOrbits< ScalarType >, std::placeholders::_1,
333  eccentricity, hyperbolicMeanAnomaly ) );
334 
335  // Add the first derivative of the root function.
336  rootFunction->addBinding(
337  -1, std::bind(
338  &computeFirstDerivativeKeplersFunctionForHyperbolicOrbits< ScalarType >,
339  std::placeholders::_1, eccentricity ) );
340 
341  // Declare initial guess.
342  ScalarType initialGuess = TUDAT_NAN;
343 
344  // Set the initial guess. Check if the default scheme is to be used or a user specified
345  // value should be used. See [Wakker, 2007] for derivations of the default values.
346  // Note that an error was detected in these starter values, as is discussed in
347  // [Musegaas,2012].
348  // !!!!!!!!!!!!! IMPORTANT !!!!!!!!!!!!!
349  // If this scheme is changed, please run a very extensive test suite. The root finder
350  // function tends to be chaotic for some very specific combinations of mean anomaly and
351  // eccentricity. Various random tests of 100.000.000 samples were done to verify the
352  // functionality of this one. [Musegaas,2012]
353  if ( useDefaultInitialGuess )
354  {
355  if ( std::abs( hyperbolicMeanAnomaly ) <
356  getFloatingInteger< ScalarType >( 6 ) * eccentricity )
357  {
358  initialGuess =
359  std::sqrt( getFloatingInteger< ScalarType >( 8 ) *
360  ( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) /
361  eccentricity ) *
362  std::sinh( getFloatingFraction< ScalarType >( 1, 3 ) * boost::math::asinh(
363  getFloatingInteger< ScalarType >( 3 ) *
364  hyperbolicMeanAnomaly /
365  ( std::sqrt( getFloatingInteger< ScalarType >( 8 ) *
366  ( eccentricity -
367  getFloatingInteger< ScalarType >( 1 ) ) /
368  eccentricity ) *
369  ( eccentricity - getFloatingInteger< ScalarType >( 1 )
370  ) ) ) );
371  }
372  else if ( hyperbolicMeanAnomaly > getFloatingInteger< ScalarType >( 6 ) * eccentricity )
373  {
374  initialGuess = ( std::log( getFloatingInteger< ScalarType >( 2 ) *
375  hyperbolicMeanAnomaly / eccentricity ) );
376  }
377  else
378  {
379  initialGuess = ( - std::log( -getFloatingInteger< ScalarType >( 2 ) *
380  hyperbolicMeanAnomaly / eccentricity ) );
381  }
382  }
383  else
384  {
385  initialGuess = userSpecifiedInitialGuess;
386  }
387 
388  // Set hyperbolic eccentric anomaly based on result of Newton-Raphson root-finding
389  // algorithm.
390  try
391  {
392  hyperbolicEccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
393 
394  }
395  catch( std::runtime_error const& )
396 
397  {
398  rootFinder = createRootFinder< ScalarType >(
399  bisectionRootFinderSettings(
400  TUDAT_NAN, 20.0 * std::numeric_limits< ScalarType >::epsilon( ),
401  TUDAT_NAN, 100 ), getFloatingInteger< ScalarType >( 0 ),
402  getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ), root_finders::accept_result );
403 
404  hyperbolicEccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
405  }
406 
407  }
408 
409  // In this case the orbit is not hyperbolic.
410  else
411  {
412  throw std::runtime_error( "Invalid eccentricity. Valid range is e > 1.0. Eccentricity was: " +
413  std::to_string( eccentricity ) );
414  }
415 
416  // Return hyperbolic eccentric anomaly.
417  return hyperbolicEccentricAnomaly;
418 }
