54 
55  // Set components of empirical accelerations that are to be estimated
56  for( std::map< basic_astrodynamics::EmpiricalAccelerationComponents,
57  std::vector< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes > >::
58  const_iterator inputComponentIterator = componentsToEstimate.begin( );
59  inputComponentIterator != componentsToEstimate.end( );
60  inputComponentIterator++ )

62  int currentComponentIndex = static_cast< int >( inputComponentIterator->first );
63 
64  // Iterate over all components
65  for( unsigned int i = 0; i < inputComponentIterator->second.size( ); i++ )
66  {
67  // Check feasibility of input
68  if( accelerationIndices_.count( inputComponentIterator->second.at( i ) ) > 0 )
69  {
70  std::vector< int > currentIndices = accelerationIndices_.at( inputComponentIterator->second.at( i ) );
71  if( std::find( currentIndices.begin( ), currentIndices.end( ), currentComponentIndex ) !=
72  currentIndices.end( ) )
73  {
74  throw std::runtime_error(
75  "Error when creating empirical acceleration parameter object, found duplicate of component." );
76  }
77  }
78  accelerationIndices_[ inputComponentIterator->second.at( i ) ].push_back( currentComponentIndex );
79  parameterSize_++;
80  }

93  Eigen::VectorXd getParameterValue( )
94  {
95  Eigen::VectorXd parameter = Eigen::VectorXd::Zero( parameterSize_ );
96 
97  // Iterate over all acceleration models
98  for( unsigned int i = 0; i < empiricalAcceleration_.size( ); i++ )
99  {
100 
101  Eigen::VectorXd currentAccelerationParameter = Eigen::VectorXd::Zero( parameterSize_ );
102 
103  // Iterate over all components and retrieve required values
104  Eigen::Matrix3d accelerationComponents = empiricalAcceleration_.at( i )->getAccelerationComponents( );
105  int currentIndex = 0;
106  for( std::map< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes, std::vector< int > >::const_iterator
107  componentIterator = accelerationIndices_.begin( );
108  componentIterator != accelerationIndices_.end( ); componentIterator++ )
109  {
110  Eigen::Vector3d accelerations =
111  accelerationComponents.block( 0, static_cast< int >( componentIterator->first ), 3, 1 );
112 
113  for( unsigned int i = 0; i < componentIterator->second.size( ); i++ )
114  {
115  currentAccelerationParameter( currentIndex ) = accelerations( componentIterator->second.at( i ) );
116  currentIndex++;
117  }
118  }
119 
120  // Check consistency
121  if( currentIndex != parameterSize_ )
122  {
123  throw std::runtime_error( "Error when getting empirical parameter; inconsistent sizes found." );
124  }
125  if( i == 0 )
126  {
127  parameter = currentAccelerationParameter;
128  }
129  else if( parameter != currentAccelerationParameter )
130  {
131  std::cerr<<"Warning when retrieving emp. acc. parameter from list. List entries are incompatible. First entry is: "<<std::endl<<
132  parameter<<std::endl<<" Current value is: "<<std::endl<<currentAccelerationParameter<<std::endl<<
133  "Difference is: "<<parameter - currentAccelerationParameter<<std::endl;
134  }
135  }
136 
137  return parameter;
138  }

156  {
157  Eigen::Vector3d accelerations =
158  accelerationComponents.block( 0, static_cast< int >( componentIterator->first ), 3, 1 );

165  }

169

312  {

314  }

317  empiricalAccelerationInterpolator_ = std::make_shared<
318  interpolators::PiecewiseConstantInterpolator< double, Eigen::Matrix3d > >(

320  typedef interpolators::OneDimensionalInterpolator< double, Eigen::Matrix3d > LocalInterpolator;
321 
322  empiricalAccelerationFunction_ =
