168  ratioOfSpecificHeats_( 1.4 ), boundaryHandling_( boundaryHandling )
169  {
170  // Assign default values
171  defaultExtrapolationValue_.resize( dependentVariablesNames.size( ) );
172  for ( unsigned int i = 0; i < dependentVariablesNames.size( ); i++ )
173  {
174  for ( unsigned int j = 0; j < independentVariablesNames.size( ); j++ )
175  {
176  if ( boundaryHandling_.at( j ) == interpolators::use_default_value ||
177  boundaryHandling_.at( j ) == interpolators::use_default_value_with_warning )
178  {
179  defaultExtrapolationValue_.at( i ).push_back( std::make_pair( defaultExtrapolationValue.at( i ),
180  defaultExtrapolationValue.at( i ) ) );
181  }
182  else
183  {
184  defaultExtrapolationValue_.at( i ).push_back( std::make_pair( IdentityElement::getAdditionIdentity< double >( ),
