120  DataType execute( const FunctionPointer rootFunction, const DataType initialGuess )
121  {
122  // Set the root function.
123  this->rootFunction = rootFunction;
124 
125  // Start at initial guess, and compute the function value, its first and second derivative.
126  DataType currentRootValue = TUDAT_NAN;
127  DataType nextRootValue = initialGuess;
128  DataType currentFunctionValue = TUDAT_NAN;
129  DataType nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
130  DataType currentFirstDerivativeValue = TUDAT_NAN;
131  DataType nextFirstDerivativeValue = this->rootFunction->
132  computeDerivative( 1, nextRootValue );
133  DataType currentSecondDerivativeValue = TUDAT_NAN;
134  DataType nextSecondDerivativeValue = this->rootFunction->
135  computeDerivative( 2, nextRootValue );
136 
137  // Loop counter.
138  unsigned int counter = 1;
139 
140  // Loop until we have a solution with sufficient accuracy.
141  do
142  {
143  // Save the old values.
144  currentRootValue = nextRootValue;
145  currentFunctionValue = nextFunctionValue;
146  currentFirstDerivativeValue = nextFirstDerivativeValue;
147  currentSecondDerivativeValue = nextSecondDerivativeValue;
148 
149  // Compute next value of root using the following algorithm (see class documentation):
150  nextRootValue = currentRootValue
151  - ( ( 2.0 * currentFunctionValue * currentFirstDerivativeValue )
152  / ( 2.0 * currentFirstDerivativeValue * currentFirstDerivativeValue
153  - currentFunctionValue * currentSecondDerivativeValue ) );
154  nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
155  nextFirstDerivativeValue = this->rootFunction->computeDerivative( 1, nextRootValue );
156  nextSecondDerivativeValue = this->rootFunction->computeDerivative( 2, nextRootValue );
157 
158  // Update the counter.
159  counter++;
160  }
161  while( nextFunctionValue != mathematical_constants::getFloatingInteger< DataType >( 0 ) &&
162  !this->terminationFunction_( nextRootValue, currentRootValue, nextFunctionValue,
163  currentFunctionValue, counter ) );
164 
165  return nextRootValue;
166  }
