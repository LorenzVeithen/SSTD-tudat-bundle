132  void updateMembers( const double currentTime = TUDAT_NAN )
133  {
134  if( !( this->currentTime_ == currentTime ) )
135  {
136  // Calculate current normal vector and distance from accelerated body to source
137  currentNormalizedVectorToSource_ = ( sourcePositionFunction_( ) - acceleratedBodyPositionFunction_( ) );
138  currentDistanceToSource_ = currentNormalizedVectorToSource_.norm( );
139  currentNormalizedVectorToSource_.normalize( );
140 
141  // Retrieve current radiation pressure and mass
142  currentRadiationPressure_ = radiationPressureFunction_( );
143  currentMass_ = massFunction_( );
144 
145  // Iterate over all panels and calculate acceleration due to radiation pressure on panel
146  currentAcceleration_.setZero( );
147  if( currentRadiationPressure_ > 0.0 )
148  {
149  for( int i = 0; i < numberOfPanels_; i++ )
150  {
151  currentPanelAccelerations_[ i ] = currentRadiationPressure_ / massFunction_( ) * computeSinglePanelNormalizedRadiationPressureForce(
152  currentNormalizedVectorToSource_ , panelSurfaceNormalFunctions_[ i ]( ), panelAreaFunctions_[ i ]( ),
153  panelEmissivittyFunctions_[ i ]( ), panelDiffuseReflectionCoefficientFunctions_[ i ]( ) );
154  currentAcceleration_ += currentPanelAccelerations_[ i ];
155  }
156  }
157  this->currentTime_ = currentTime;
158  }
159  }
