146 class ScaledAtmosphereModel: public AtmosphereModel
147 {
148 public:
149  ScaledAtmosphereModel(
150  const std::shared_ptr< AtmosphereModel > baseAtmosphere,
151  const std::function< double( const double ) > densityScalingFunction,
152  const bool isScalingAbsolute = true ):
153  AtmosphereModel( ),
154  baseAtmosphere_( baseAtmosphere ),
155  densityScalingFunction_( densityScalingFunction ),
156  isScalingAbsolute_( isScalingAbsolute ){ }
157 
158  double getDensity( const double altitude, const double longitude,
159  const double latitude, const double time )
160  {
161  if( isScalingAbsolute_ )
162  {
163  return baseAtmosphere_->getDensity( altitude, longitude, latitude, time ) +
164  densityScalingFunction_( time );
165  }
166  else
167  {
168  return baseAtmosphere_->getDensity( altitude, longitude, latitude, time ) *
169  densityScalingFunction_( time );
170  }
171  }
172

174  double getPressure( const double altitude, const double longitude,

188  double getSpeedOfSound( const double altitude, const double longitude,

190  {

192  }
