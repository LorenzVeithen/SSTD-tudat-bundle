142  virtual void updateCurrentCoefficients( const std::vector< double >& independentVariables,
143  const double currentTime = TUDAT_NAN )
144  {
145  // Check if the correct number of aerodynamic coefficients is provided.
146  if( independentVariables.size( ) != numberOfIndependentVariables_ )
147  {
148  throw std::runtime_error(
149  "Error in CustomAerodynamicCoefficientInterface, number of input variables is inconsistent " +
150  std::to_string( independentVariables.size( ) ) + ", " +
151  std::to_string( numberOfIndependentVariables_ ) );
152  }
153 
154  // Update current coefficients.
155  if( timeUpdateFunction_ != nullptr )
156  {
157  timeUpdateFunction_( currentTime );
158  }
159 
160  Eigen::Vector6d currentCoefficients = coefficientFunction_(
161  independentVariables );
162  currentForceCoefficients_ = currentCoefficients.segment( 0, 3 );
163  currentMomentCoefficients_ = currentCoefficients.segment( 3, 3 );
164  }
