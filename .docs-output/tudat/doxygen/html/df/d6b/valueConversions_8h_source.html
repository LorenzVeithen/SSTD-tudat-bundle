59 template< template< typename ... > class MapType, typename KeyType, typename ValueType >
60 MapType< KeyType, ValueType > mapFromJson( const nlohmann::json& jsonObject )
61 {
62  MapType< KeyType, ValueType > map;
63  for ( nlohmann::json::const_iterator it = jsonObject.begin( ); it != jsonObject.end( ); ++it )
64  {
65  const std::string key = it.key( );
66  if ( ! contains( SpecialKeys::all, key ) )
67  {
68  map[ boost::lexical_cast< KeyType >( key ) ] = getValue< ValueType >( jsonObject, key );
69  }
70  }
71  return map;
72 }

198 template< typename ScalarType, int rows, int cols >
199 void from_json( const nlohmann::json& jsonObject, Matrix< ScalarType, rows, cols >& matrix )
200 {
201  nlohmann::json jsonArray;
202  if ( jsonObject.is_array( ) )
203  {
204  if ( jsonObject.empty( ) )
205  {
206  return;
207  }
208  jsonArray = jsonObject;
209  }
210  else if ( jsonObject.is_number( ) )
211  {
212  jsonArray.push_back( jsonObject );
213  }
214  else
215  {
216  throw nlohmann::detail::type_error::create( 0, "" );
217  }
218 
219  nlohmann::json jsonArrayOfArrays;
220  if ( jsonArray.front( ).is_array( ) ) // provided matrix
221  {
222  if( jsonArray.at( 0 ).size( ) == 1 )
223  {
224  for ( unsigned int i = 0; i < jsonArray.size( ); ++i )
225  {
226  jsonArrayOfArrays.push_back( { jsonArray.at( i ).at( 0 ) } );
227  }
228  }
229  else
230  {
231  jsonArrayOfArrays = jsonArray;
232  }
233  }
234  else // provided vector
235  {
236  if ( rows == 1 ) // expected row vector
237  {
238  jsonArrayOfArrays.push_back( jsonArray );
239  }
240  else if ( cols == 1 ) // expected column vector
241  {
242  for ( unsigned int i = 0; i < jsonArray.size( ); ++i )
243  {
244  jsonArrayOfArrays.push_back( { jsonArray.at( i ) } );
245  }
246  }
247  else // expected matrix
248  {
249  std::cerr << "Expected a matrix, received a vector." << std::endl;
250  throw nlohmann::detail::type_error::create( 0, "" );
251  }
252  }
253 
254  const unsigned int providedRows = jsonArrayOfArrays.size( );
255  const unsigned int providedCols = jsonArrayOfArrays.front( ).size( );
256  if ( ( rows >= 0 && int( providedRows ) != rows ) || ( cols >= 0 && int( providedCols ) != cols ) )
257  {
258  std::cerr << "Expected matrix of size " << rows << "x" << cols
259  << ", received matrix of size " << providedRows << "x" << providedCols << "." << std::endl;
260  throw nlohmann::detail::type_error::create( 0, "" );
261  }
262 
263  matrix.resize( providedRows, providedCols );
264  for ( unsigned int r = 0; r < providedRows; ++r )
265  {
266  if ( jsonArrayOfArrays.at( r ).size( ) != providedCols )
267  {
268  std::cerr << "Unconsistent matrix size: some rows have different number of columns." << std::endl;
269  throw nlohmann::detail::type_error::create( 0, "" );
270  }
271  for ( unsigned int c = 0; c < providedCols; ++c )
272  {
273  if( providedCols > 1 && providedRows > 1 )
274  {
275  matrix( r, c ) = jsonArrayOfArrays.at( r ).at( c );
276  }
277  else if( providedCols == 1 )
278  {
279  matrix( r, c ) = jsonArrayOfArrays.at( r );
280  }
281  else if( providedRows == 1 )
282  {
283  matrix( r, c ) = jsonArrayOfArrays.at( c );
284  }
285  }
286  }
