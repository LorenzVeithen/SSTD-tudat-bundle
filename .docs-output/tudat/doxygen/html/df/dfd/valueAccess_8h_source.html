187 template< typename ValueType >
188 ValueType getValue( const nlohmann::json& jsonObject, const KeyPath& keyPath )
189 {
190  nlohmann::json rootObject;
191  nlohmann::json currentObject = jsonObject;
192  KeyPath currentKeyPath = keyPath;
193  KeyPath canonicalKeyPath = keyPath;
194 
195  if ( ! contains( currentKeyPath, SpecialKeys::keyPath ) )
196  {
197  canonicalKeyPath = currentKeyPath.canonical( getKeyPath( currentObject ) );
198 
199  if ( ! contains( currentKeyPath, SpecialKeys::rootObject ) )
200  {
201  if ( isDefined( currentObject, SpecialKeys::rootObject ) )
202  {
203  rootObject = getValue< nlohmann::json >( currentObject, SpecialKeys::rootObject );
204  }
205 
206  if ( currentKeyPath.size( ) > 0 )
207  {
208  if ( currentKeyPath.front( ) == SpecialKeys::root || contains( currentKeyPath, SpecialKeys::up ) )
209  {
210  // Path is absolute or contains up-key
211 
212  // Use absolute key path
213  currentKeyPath = canonicalKeyPath;
214 
215  // Use jsonObject's root object instead
216  if ( rootObject.is_null( ) )
217  {
218  throw UndefinedKeyError( canonicalKeyPath );
219  }
220  currentObject = rootObject;
221  }
222  }
223  }
224  }
225 
226  if ( currentKeyPath.isAbsolute( ) )
227  {
228  // Remove "~"
229  currentKeyPath.erase( currentKeyPath.begin( ) );
230  }
231 
232  accessedKeyPaths.insert( canonicalKeyPath );
233 
234  try
235  {
236  currentObject = valueAt( currentObject, currentKeyPath );
237  }
238  catch ( ... )
239  {
240  // At least one of the keys was missing
241  throw UndefinedKeyError( canonicalKeyPath );
242  }
243 
244  if ( ! containsAnyOf( currentKeyPath, SpecialKeys::all ) )
245  {
246  // If jsonObject is an array containing objects, convert to object
247  convertToObjectIfContainsObjects( currentObject );
248 
249  if ( currentObject.is_object( ) )
250  {
251  // Define keys rootObject and keyPath of jsonObject to be returned
252  currentObject[ SpecialKeys::rootObject ] = rootObject.is_null( ) ? jsonObject : rootObject;
253  currentObject[ SpecialKeys::keyPath ] = canonicalKeyPath;
254  }
255  }
256 
257  try
258  {
259  // Try to convert to the requested type
260  const ValueType convertedObject = currentObject;
261  return convertedObject;
262  }
263  catch ( const nlohmann::detail::type_error& )
264  {
265  // Could not convert to the requested type
266  throw IllegalValueError( canonicalKeyPath, currentObject, typeid( ValueType ) );
267  }
268  catch ( const UnknownEnumError& )
269  {
270  std::cout<<"Test A"<<std::endl;
271  // Could not convert string to enum
272  throw IllegalValueError( canonicalKeyPath, currentObject, typeid( ValueType ) );
273  }
274  catch ( const UndefinedKeyError& error )
275  {
276  // Some of the keys needed during creation of object of type ValueType were missing
277  throw error;
278  }
279  catch ( const IllegalValueError& error )
280  {
281  // Some of the values for the keys needed during creation of object of type ValueType were illegal
282  throw error;
283  }
284 }

346 template< typename ValueType >
347 ValueType getValue( const nlohmann::json& jsonObject, const std::vector< KeyPath >& keyPaths )
348 {
349  for ( unsigned int i = 0; i < keyPaths.size( ); ++i )
350  {
351  try
352  {
353  return getValue< nlohmann::json >( jsonObject, keyPaths.at( i ) );
354  }
355  catch ( const UndefinedKeyError& error )
356  {
357  error.rethrowIfNotTriggeredByMissingValueAt( keyPaths.at( i ) );
358  if ( i == keyPaths.size( ) - 1 )
359  {
360  throw error;
361  }
362  }
363  }
364  throw; // silence warning may reach end of non-void function
365 }
