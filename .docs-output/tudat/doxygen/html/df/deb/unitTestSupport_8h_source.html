69 template< typename Enum >
70 void checkConsistentEnum( const std::string& filename,
71  const std::map< Enum, std::string >& stringValues,
72  const std::vector< Enum >& usupportedValues )
73 {
74  using namespace json_interface;
75 
76  // Create vector of supported values
77  std::vector< Enum > supportedValues;
78  for ( auto entry : stringValues )
79  {
80  Enum value = entry.first;
81  if ( ! contains( usupportedValues, value ) )
82  {
83  supportedValues.push_back( value );
84  }
85  }
86 
87  // Check that values and supportedValues are equivalent
88  const std::vector< Enum > values = parseJSONFile< std::vector< Enum > >( filename );
89  BOOST_CHECK( containsAllOf( values, supportedValues ) && containsAllOf( supportedValues, values ) );
90 }

95 template< typename ContainerType >
96 void checkCloseIntegrationResultsMatrix( const std::map< double, ContainerType >& results1,
97  const std::map< double, ContainerType >& results2,
98  const std::vector< std::pair< unsigned int, unsigned int > > indices,
99  const std::vector< std::pair< unsigned int, unsigned int > > sizes,
100  const std::vector< double > absoluteTolerances )
101 {
102  // Check size of maps
103  BOOST_CHECK_EQUAL( results1.size( ), results2.size( ) );
104 
105  auto it1 = results1.begin( );
106  auto it2 = results2.begin( );
107 
108  for( unsigned int i = 0; i < results1.size( ); i++ )
109  {
110  BOOST_CHECK_SMALL( std::fabs( it1->first - it2->first ),
111  2.0 * it1->first * std::numeric_limits< double >::epsilon( ) );
112  const Eigen::MatrixXd state1 = it1->second;
113  const Eigen::MatrixXd state2 = it2->second;
114 
115  for ( unsigned int i = 0; i < indices.size( ); ++i )
116  {
117  for( unsigned int row = 0; row < sizes.at( i ).first; row++ )
118  {
119  for( unsigned int col = 0; col < sizes.at( i ).second; col++ )
120  {
121  BOOST_CHECK_SMALL(
122  std::fabs(
123  state1( indices.at( i ).first + row, indices.at( i ).second + col ) -
124  state2( indices.at( i ).first + row, indices.at( i ).second + col ) ), absoluteTolerances.at( i ) );
125  }
126  }
127  }
128 
129  }
130 }
