163 template <typename OutputStream, typename ScalarType, int NumberOfRows,
164  int NumberOfColumns, int Options, int MaximumRows, int MaximumCols>
165 void writeValueToStream(
166  OutputStream &stream,
167  const Eigen::Matrix<ScalarType, NumberOfRows, NumberOfColumns, Options,
168  MaximumRows, MaximumCols> &value,
169  const int precision, const std::string &delimiter,
170  const bool endLineAfterRow = 0) {
171  for (int i = 0; i < value.rows(); i++) {
172  for (int j = 0; j < value.cols(); j++) {
173  stream << delimiter << " " << std::setprecision(precision) << std::left
174  << std::setw(precision + 1) << value(i, j);
175  }
176  if (endLineAfterRow) {
177  stream << std::endl;
178  }
179  }
180  stream << std::endl;
181 }

217 template <typename InputIterator>
218 void writeDataMapToTextFile(InputIterator iteratorDataMap, InputIterator last,
219  const std::string &outputFilename,
220  const boost::filesystem::path &outputDirectory,
221  const std::string &fileHeader,
222  const int precisionOfKeyType,
223  const int precisionOfValueType,
224  const std::string &delimiter) {
225  // Check if output directory exists; create it if it doesn't.
226  if (!boost::filesystem::exists(outputDirectory)) {
227  boost::filesystem::create_directories(outputDirectory);
228  }
229 
230  // Open output file.
231  std::string outputDirectoryAndFilename =
232  outputDirectory.string() + "/" + outputFilename;
233  std::ofstream outputFile_(outputDirectoryAndFilename.c_str());
234 
235  // Write file header to file.
236  outputFile_ << fileHeader;
237 
238  // Loop over map of propagation history.
239  for (; iteratorDataMap != last; iteratorDataMap++) {
240  // Print map data to output file.
241  outputFile_ << std::setprecision(precisionOfKeyType) << std::left
242  << std::setw(precisionOfKeyType + 1) << iteratorDataMap->first;
243  writeValueToStream(outputFile_, iteratorDataMap->second,
244  precisionOfValueType, delimiter);
245  }
246 
247  // Close output file.
248  outputFile_.close();
249 }
