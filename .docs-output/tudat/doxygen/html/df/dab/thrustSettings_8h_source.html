669 // Interface base class that can be defined by user to make the use of the ParameterizedThrustMagnitudeSettings class easier
679 class ThrustInputParameterGuidance
680 {
681 public:
682 
683  // Constructor
693  ThrustInputParameterGuidance(
694  const int numberOfThrustInputParameters,
695  const int numberOfSpecificImpulseInputParameters,
696  const bool includeThrottleSetting = false,
697  const int throttleSettingIndex = -1 ):
698  numberOfThrustInputParameters_( numberOfThrustInputParameters ),
699  numberOfSpecificImpulseInputParameters_( numberOfSpecificImpulseInputParameters ),
700  includeThrottleSetting_( includeThrottleSetting ), throttleSettingIndex_( throttleSettingIndex )
701  {
702  if( includeThrottleSetting && ( throttleSettingIndex < 0 || throttleSettingIndex >= numberOfThrustInputParameters ) )
703  {
704  throw std::runtime_error( "Error when creating ThrustInputParameterGuidance, inconsistent throttle input" );
705  }
706  else if( !includeThrottleSetting && throttleSettingIndex >= 0 )
707  {
708  throw std::runtime_error( "Error when creating ThrustInputParameterGuidance, no throttle input with positive throttle index" );
709  }
710 
711  currentThrustGuidanceParameters_.resize( numberOfThrustInputParameters_ );
712  currentSpecificImpulseParameters_.resize( numberOfSpecificImpulseInputParameters_ );
713  }
714 
715  // Destructor
716  virtual ~ThrustInputParameterGuidance( ){ }
717 
718  // Function to retrieve the number of guidance-input parameters that are computed for the thrust
723  int getNumberOfThrustInputParameters( )
724  {
725  return numberOfThrustInputParameters_;
726  }
727 
728  // Function to retrieve the number of guidance-input parameters that are computed for the specific impulse
733  int getNumberOfSpecificImpulseInputParameters( )
734  {
735  return numberOfSpecificImpulseInputParameters_;
736  }
737 
738  // Function to retrieve a guidance-input parameter that for the thrust
744  double getThrustInputGuidanceParameter( const int inputParameterIndex )
745  {
746  return currentThrustGuidanceParameters_.at( inputParameterIndex );
747  }
748 
749  // Function to retrieve a guidance-input parameter that for the specific impulse
755  double getSpecificImpulseInputGuidanceParameter( const int inputParameterIndex )
756  {
757  return currentSpecificImpulseParameters_.at( inputParameterIndex );
758  }
759 
760  // Function to retrieve whether any of the guidance-input parameters for the thrust define a throttle setting
765  bool getIncludeThrottleSetting( )
766  {
767  return includeThrottleSetting_;
768  }
769 
770  // Function to retrieve the index in list of thrust guidance parameters that denotes the throttle setting
775  int getThrottleSettingIndex( )
776  {
777  return throttleSettingIndex_;
778  }
779 
780  // Pure virtual function that updates the guidance algorithm to the current time/state
781  virtual void updateGuidanceParameters( ) = 0;
782 
783  // Update function for the guidance algorithm
789  void update( const double time )
790  {
791  if( time != time )
792  {
793  currentTime_ = time;
794  }
795  else
796  {
797  if( !( currentTime_ == time ) )
798  {
799  currentTime_ = time;
800  updateGuidanceParameters( );
801  }
802  }
803  }
804 
805 protected:
806 
807  // Vector of guidance-input parameters that are computed for the thrust
808  std::vector< double > currentThrustGuidanceParameters_;
809 
810  // Vector of guidance-input parameters that are computed for the specific impulse
811  std::vector< double > currentSpecificImpulseParameters_;
812 
813  // Number of guidance-input parameters that are computed for the thrust
814  int numberOfThrustInputParameters_;
815 
816  // Number of guidance-input parameters that are computed for the specific impulse
817  int numberOfSpecificImpulseInputParameters_;
818 
819  // Boolean that denotes whether any of the guidance-input parameters for the thrust define a throttle setting
820  bool includeThrottleSetting_;
821 
822  // Index in list of thrust guidance parameters that denotes the throttle setting (if any)
823  int throttleSettingIndex_;
824 
825  // Current time of interface object (e.g. time to which object was last updated).
826  double currentTime_;
827 
828 };

830 // Class to compute throttling law for the parameterized thrust magnitude, where the throttle is determined from a maximum
831 // axial g-load.
839 class AccelerationLimitedThrottleGuidance: public ThrustInputParameterGuidance
840 {
841 public:
842 
843  // Constructor.
855  AccelerationLimitedThrottleGuidance(
856  const SystemOfBodies& bodies,
857  const std::string nameOfBodyWithGuidance,
858  const std::string nameOfCentralBody,
859  const std::vector< propulsion::ThrustIndependentVariables > independentVariables,
860  const std::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator,
861  const double maximumAcceleration ): ThrustInputParameterGuidance( 1, 0, true, 0 ),
862  bodyWithGuidance_( bodies.at( nameOfBodyWithGuidance ) ), thrustInterpolator_( thrustInterpolator ),
863  maximumAcceleration_( maximumAcceleration )
864  {
865  // Split independent variables into environmental/guidance.
866  int numberOfThrottles = 0;
867  std::vector< propulsion::ThrustIndependentVariables > guidanceFreeIndependentVariables;
868  for( unsigned int i = 0; i < independentVariables.size( ); i++ )
869  {
870  if( independentVariables.at( i ) == propulsion::throttle_dependent_thrust )
871  {
872  numberOfThrottles ++;
873  }
874  else if( independentVariables.at( i ) == propulsion::guidance_input_dependent_thrust )
875  {
876  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, guidance input not supported" );
877  }
878  else
879  {
880  guidanceFreeIndependentVariables.push_back( independentVariables.at( i ) );
881  }
882  }
883 
884  // Check input consistency.
885  if( numberOfThrottles > 1 )
886  {
887  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, multiple throttles detected" );
888  }
889 
890  // Create thrust input.
891  if( bodyWithGuidance_->getFlightConditions( ) == nullptr && nameOfCentralBody != "" )
892  {
893  bodyWithGuidance_->setFlightConditions(
894  createAtmosphericFlightConditions( bodyWithGuidance_,
895  bodies.at( nameOfCentralBody ),
896  nameOfBodyWithGuidance,
897  nameOfCentralBody ) );
898  }
899  thrustInputFunctions_ = getPropulsionInputVariables(
900  bodyWithGuidance_, guidanceFreeIndependentVariables );
901 
902  currentThrustInput_.resize( thrustInputFunctions_.size( ) );
903  }

905  // Function that updates the guidance algorithm to the current time/state: sets the throttle value based on axia
910  void updateGuidanceParameters( )
911  {
912  // Compute environmental input variables for thrust.
913  for( unsigned int i = 0; i < thrustInputFunctions_.size( ); i++ )
914  {
915  currentThrustInput_[ i ] = thrustInputFunctions_.at( i )( );
916  }
917 
918  // Compute maximum thrust
919  double currentThrust = thrustInterpolator_->interpolate( currentThrustInput_ );
920  double currentMass = bodyWithGuidance_->getBodyMass( );
921 
922  // Set throttle (to < 1 if necessary)
923  if( currentThrust / currentMass < maximumAcceleration_ )
924  {
925  currentThrustGuidanceParameters_[ 0 ] = 1.0;
926  }
927  else
928  {
929  currentThrustGuidanceParameters_[ 0 ] = maximumAcceleration_ * currentMass / currentThrust;
930  }
931  }
