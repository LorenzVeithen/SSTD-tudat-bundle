163  double getPressure( const double altitude, const double longitude = 0.0,
164  const double latitude = 0.0, const double time = 0.0 )
165  {
166  if( constantTemperature_ != constantTemperature_ )
167  {
168  throw std::runtime_error( "Error, exponential atmosphere temperature not defined" );

176  TUDAT_UNUSED_PARAMETER( latitude );
177  TUDAT_UNUSED_PARAMETER( time );
178  return getDensity( altitude ) * specificGasConstant_ * constantTemperature_;
179  }
