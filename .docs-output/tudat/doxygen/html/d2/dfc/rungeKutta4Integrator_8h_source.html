115  virtual StateType performIntegrationStep( const TimeStepType stepSize )
116  {
117  lastIndependentVariable_ = currentIndependentVariable_;
118  lastState_ = currentState_;
119 
120  // Calculate k1-k4.
121  StateDerivativeType k1, k2, k3, k4;
122  for ( unsigned int i = 1; i <= 4; i++ )
123  {
124  IndependentVariableType time;
125  StateType state;
126  switch ( i )
127  {
128  case 1:
129  time = currentIndependentVariable_;
130  state = currentState_;
131  k1 = stepSize * this->stateDerivativeFunction_( time, state );
132  break;
133  case 2:
134  time = currentIndependentVariable_ + stepSize / 2.0;
135  state = static_cast< StateType >( currentState_ + k1 / 2.0 );
136  k2 = stepSize * this->stateDerivativeFunction_( time, state );
137  break;
138  case 3:
139  time = currentIndependentVariable_ + stepSize / 2.0;
140  state = static_cast< StateType >( currentState_ + k2 / 2.0 );
141  k3 = stepSize * this->stateDerivativeFunction_( time, state );
142  break;
143  case 4:
144  time = currentIndependentVariable_ + stepSize;
145  state = static_cast< StateType >( currentState_ + k3 );
146  k4 = stepSize * this->stateDerivativeFunction_( time, state );
147  break;
148  }
149 
150  // Check if propagation should terminate because the propagation termination condition has been reached
151  // while computing k1, k2, k3 or k4. If so, return immediately the current state (not recomputed yet),
152  // which will be discarded.
153  if ( this->propagationTerminationFunction_( static_cast< double >( time ), TUDAT_NAN ) )
154  {
155  this->propagationTerminationConditionReachedDuringStep_ = true;
156  return currentState_;
157  }
158  }
159 
160  stepSize_ = stepSize;
161  currentIndependentVariable_ += stepSize_;
162  currentState_ += ( k1 + 2.0 * k2 + 2.0 * k3 + k4 ) / 6.0;
163 
164  // Return the integration result.
165  return currentState_;
166  }
