300 template< typename AngleScalarType = double >
301 Eigen::Quaternion< AngleScalarType > getRotatingPlanetocentricToLocalVerticalFrameTransformationQuaternion(
302  const AngleScalarType longitude, const AngleScalarType latitude )
303 {
304  // Compute transformation quaternion.
305  // Note the sign change, because how angleAxisd is defined.
306  Eigen::AngleAxis< AngleScalarType > RotationAroundZaxis = Eigen::AngleAxis< AngleScalarType >(
307  -longitude, Eigen::Matrix< AngleScalarType, 3, 1 >::UnitZ( ) );
308  Eigen::AngleAxis< AngleScalarType > RotationAroundYaxis = Eigen::AngleAxis< AngleScalarType >(
309  -( -latitude - mathematical_constants::getPi< AngleScalarType >( ) /
310  mathematical_constants::getFloatingInteger< AngleScalarType >( 2 ) ),
311  Eigen::Matrix< AngleScalarType, 3, 1 >::UnitY( ) );
312  Eigen::Quaternion< AngleScalarType > frameTransformationQuaternion = Eigen::Quaternion< AngleScalarType >(
313  ( RotationAroundYaxis * RotationAroundZaxis ) );
314 
315  // Return transformation quaternion.
316  return frameTransformationQuaternion;
317 }
